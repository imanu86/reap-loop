def is_palindrome(s: str) -> bool:
    """Return True if the string is a palindrome ignoring case and spaces."""
    cleaned = ''.join(c.lower() for c in s if c != ' ')
    return cleaned == cleaned[::-1]


def word_count(text: str) -> dict:
    """Return a dict mapping each lowercase word to its count."""
    words = text.lower().split()
    counts = {}
    for word in words:
        counts[word] = counts.get(word, 0) + 1
    return counts


def fizzbuzz(n: int) -> list:
    """Return the list of FizzBuzz strings from 1 to n."""
    result = []
    for i in range(1, n + 1):
        if i % 3 == 0 and i % 5 == 0:
            result.append("FizzBuzz")
        elif i % 3 == 0:
            result.append("Fizz")
        elif i % 5 == 0:
            result.append("Buzz")
        else:
            result.append(str(i))
    return result
