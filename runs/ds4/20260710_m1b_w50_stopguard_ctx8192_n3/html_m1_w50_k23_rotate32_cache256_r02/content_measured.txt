        btn.addEventListener('click', function (e) {

            let error = '';
            if nome.value.trim() === '' || email.value.trim() === '' || message.value.trim() === '' || error = 'Compia tutti campi';
            if error) {
                errorDiv.textContent = error;
                errorDiv.style.visibility = 'block';
                return;
            }
            if email.value.indexOf('@') === -1 || email.value.indexOf('.') === -1) {
                error = 'Inserisci una email valida';
                errorDiv.textContent = error;
                errorDiv.style.visibility = 'block';
                return;
            }
            errorDiv.style.visibility = 'hidden';
            popupOver.classList.add('show');
            // form.reset();
        }

        closeBtn.addEventListener('click', function () {
            popupOver.classList.remove('show');
        });

        window.addEventListener('click', function (e) {
            if e.target === popupOver) {
                popupOver.classList.remove('show');
