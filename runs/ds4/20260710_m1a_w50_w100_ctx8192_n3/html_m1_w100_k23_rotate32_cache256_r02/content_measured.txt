    <script>
    // JS per popup e modulo
    function() {
        // Previsti errori e invio
        document.getElementById('contactForm'). addEventListener('submit', function(e) {
            e.preventDefault(); // blocco invio trad
            // Validazione base
            const name = document.getElementById('name);
            const email = document.getElementById('email);
            const message = document.getElementById('message);
            if (name.value.trim() === '' || email.value.trim() === '' || message.value.trim() === '' ) {
                alert('c' da compire');
                return;
            }
            // Mostra popup
            document.getElementById('popupOver'). classList.add('show');
            // Reset eventuale
            // Non serve per pulire
        });

        // Chiudi popup
        document.getElementById('closePopup'). addEventListener('click', function() {
            document.getElementById('popupOver'). classList.remove('show');
        });

        // Chiudi overlay se clicchi fuori
        document.getElementById('popupOver'). addEventListener('click', function(e) {
            if (e.target.id === 'popupOver') {
                document.getElementById('popupOver'). classList.remove('show');
            }
        });
    // Prev alert per compire
