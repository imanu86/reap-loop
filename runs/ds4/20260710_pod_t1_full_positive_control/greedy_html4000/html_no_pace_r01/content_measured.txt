    <script>
        (function() {
            const form = document.getElementById('contactForm');
            const popup = document.getElementById('popupOverlay');
            const closeBtn = document.getElementById('closePopup');

            // mostra popup
            function showPopup() {
                popup.style.display = 'flex';
            }

            // nascondi popup
            function hidePopup() {
                popup.style.display = 'none';
            }

            // gestione invio form
            form.addEventListener('submit', function(e) {
                e.preventDefault();

                // piccola validazione base
                const name = document.getElementById('name').value.trim();
                const email = document.getElementById('email').value.trim();
                const message = document.getElementById('message').value.trim();

                if (!name || !email || !message) {
                    alert('⚠️ Compila tutti i campi, operatore.');
                    return;
                }

                // simula invio e mostra popup
                showPopup();

                // reset opzionale
                form.reset();
            });

            // chiusura popup
            closeBtn.addEventListener('click', hidePopup);

            // click fuori dal popup? (opzionale)
            popup.addEventListener('click', function(e) {
                if (e.target === popup) {
                    hidePopup();
                }
            });

            // tasto ESC per chiudere
            document.addEventListener('keydown', function(e) {
                if (e.key === 'Escape' && popup.style.display === 'flex') {
                    hidePopup();
                }
            });
        })();
    </script>

### Caratteristiche principali

- **Stile cyberpunk** con colori neon, bordi luminosi, sfondo scuro e font futuristico "Orbitron".
- **Tre prodotti** in griglia con pulsanti "Acquista" (simulano aggiunta al sistema).
- **Modulo contatti** completo (nome, email, messaggio) con validazione base.
- **Popup JS** che appare all'invio del modulo con il testo "Richiesta inviata" e pulsante di chiusura.
- **Codice compatto e valido** in un unico file HTML, con CSS e JS integrati.

Il popup si attiva solo dopo aver compilato tutti i campi del modulo e cliccato "Invia richiesta". Puoi chiuderlo con il pulsante "CHIUDI", cliccando fuori dalla finestra o premendo ESC.
