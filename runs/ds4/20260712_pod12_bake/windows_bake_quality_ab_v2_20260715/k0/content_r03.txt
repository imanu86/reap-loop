<script>
// ========== dati iniziali ==========
let jobs = [
  { id: 'j1', name: 'frontend-ci', status: 'success', lastBuild: '2025-03-17 11:23' },
  { id: 'j2', name: 'backend-tests', status: 'fail', lastBuild: '2025-03-17 10:15' },
  { id: 'j3', name: 'mobile-release', status: 'pending', lastBuild: '2025-03-16 22:04' },
  { id: 'j4', name: 'docs-build', status: 'success', lastBuild: '2025-03-17 09:44' }
];
let nextId = 5;

// modali
let addModal = document.getElementById('addModal');
let confirmModal = document.getElementById('confirmModal');
let confirmCallback = null;

// helper per aggiornare hero status
function updateHeroStatus() {
  const failed = jobs.filter(j => j.status === 'fail').length;
  const pending = jobs.filter(j => j.status === 'pending').length;
  let badge = document.getElementById('heroStatus');
  if (failed > 0) {
    badge.textContent = '⚠️ ' + failed + ' fallita';
    badge.style.background = '#e74c3c';
  } else if (pending > 0) {
    badge.textContent = '⏳ ' + pending + ' in attesa';
    badge.style.background = '#f39c12';
  } else {
    badge.textContent = '✅ Tutto operativo';
    badge.style.background = '#2ecc71';
  }
}

// render tabella con filtro
function renderTable() {
  const filterVal = document.getElementById('filterInput').value.toLowerCase();
  const tbody = document.getElementById('jobBody');
  tbody.innerHTML = '';
  const filtered = jobs.filter(j => j.name.toLowerCase().includes(filterVal));
  filtered.forEach(job => {
    const tr = document.createElement('tr');
    let statusClass = 'status-success';
    if (job.status === 'fail') statusClass = 'status-fail';
    else if (job.status === 'pending') statusClass = 'status-pending';
    tr.innerHTML = `
      <td class="job-name">${job.name}</td>
      <td><span class="status-badge ${statusClass}">${job.status}</span></td>
      <td>${job.lastBuild}</td>
      <td><button class="action-btn" data-id="${job.id}">🔄 Rilancia</button></td>
    `;
    tbody.appendChild(tr);
  });
  // assegna eventi rilancio
  document.querySelectorAll('#jobBody .action-btn').forEach(btn => {
    btn.addEventListener('click', function(e) {
      const id = this.dataset.id;
      const job = jobs.find(j => j.id === id);
      if (!job) return;
      // mostra popup conferma
      document.getElementById('confirmMessage').textContent = `Rilancia build per "${job.name}"?`;
      confirmModal.style.display = 'flex';
      // salva callback
      confirmCallback = function() {
        // aggiorna stato e lastBuild
        job.status = 'pending';
        job.lastBuild = new Date().toLocaleString('it-IT', { year:'numeric', month:'2-digit', day:'2-digit', hour:'2-digit', minute:'2-digit' });
        renderTable();
        updateHeroStatus();
        closeConfirmModal();
      };
      document.getElementById('confirmBtn').onclick = confirmCallback;
    });
  });
}

// filtro
function applyFilter() { renderTable(); }

// modali add
function openAddModal() { addModal.style.display = 'flex'; }
function closeAddModal() { addModal.style.display = 'none'; }
function confirmAddJob() {
  const nameInput = document.getElementById('newJobName');
  const statusSelect = document.getElementById('newJobStatus');
  const name = nameInput.value.trim();
  if (!name) { alert('Inserisci un nome job'); return; }
  const status = statusSelect.value;
  jobs.push({
    id: 'j' + (nextId++),
    name: name,
    status: status,
    lastBuild: new Date().toLocaleString('it-IT', { year:'numeric', month:'2-digit', day:'2-digit', hour:'2-digit', minute:'2-digit' })
  });
  renderTable();
  updateHeroStatus();
  nameInput.value = '';
  closeAddModal();
}

// modali confirm
function closeConfirmModal() {
  confirmModal.style.display = 'none';
  confirmCallback = null;
}

// inizializza
renderTable();
updateHeroStatus();

// chiudi modali su overlay click
document.addEventListener('click', function(e) {
  if (e.target.classList.contains('modal-overlay')) {
    closeAddModal();
    closeConfirmModal();
  }
});
</script>
