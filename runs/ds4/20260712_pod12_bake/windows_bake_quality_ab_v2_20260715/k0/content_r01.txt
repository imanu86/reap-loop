<script>
(function(){
  // stato
  const tbody = document.getElementById('tableBody');
  const filterInput = document.getElementById('filterInput');
  const resetBtn = document.getElementById('resetFilter');
  const addBtn = document.getElementById('addJobBtn');
  const newName = document.getElementById('newJobName');
  const newStatus = document.getElementById('newJobStatus');
  // popup
  const overlay = document.getElementById('popupOverlay');
  const popupTitle = document.getElementById('popupTitle');
  const popupMsg = document.getElementById('popupMsg');
  const cancelBtn = document.getElementById('popupCancel');
  const confirmBtn = document.getElementById('popupConfirm');

  let currentJob = null; // per popup

  // helper badge
  function badge(status){
    const s = status.toLowerCase().trim();
    if(s==='ok'||s==='success') return '<span class="status-badge ok">OK</span>';
    if(s==='fail'||s==='error'||s==='failed') return '<span class="status-badge fail">FAIL</span>';
    return '<span class="status-badge pending">pending</span>';
  }

  // aggiorna hero (conta)
  function updateHero(){
    const rows = tbody.querySelectorAll('tr');
    let ok=0, fail=0, pending=0;
    rows.forEach(tr=>{
      const html = tr.innerHTML;
      if(html.includes('status-badge ok')) ok++;
      else if(html.includes('status-badge fail')) fail++;
      else pending++;
    });
    document.querySelector('.hero .status').textContent = `✅ ${ok} OK`;
    document.querySelector('.hero .bad-count').textContent = `❌ ${fail} FAIL`;
    // pending badge aggiornato
    const pendingSpan = document.querySelector('.hero span:nth-child(3)');
    if(pendingSpan) pendingSpan.textContent = `⏳ ${pending} pending`;
  }

  // aggiungi riga
  function addRow(job, status, commit='', durata=''){
    const tr = document.createElement('tr');
    const statusBadge = badge(status);
    const commitTxt = commit || (Math.random().toString(36).slice(2,8));
    const durataTxt = durata || (status.toLowerCase().trim()==='pending'?'--':(Math.floor(Math.random()*90)+10)+'s');
    tr.innerHTML = `<td class="job-name">${job}</td><td>${statusBadge}</td><td>${commitTxt}</td><td>${durataTxt}</td><td><button class="action-btn" data-job="${job}">▶ rilancia</button></td>`;
    tbody.appendChild(tr);
    updateHero();
    applyFilter();
  }

  // filtro
  function applyFilter(){
    const val = filterInput.value.toLowerCase().trim();
    const rows = tbody.querySelectorAll('tr');
    rows.forEach(tr=>{
      const text = tr.textContent.toLowerCase();
      if(!val || text.includes(val)){
        tr.style.display='';
      } else {
        tr.style.display='none';
      }
    });
  }

  // reset filtro
  function resetFilter(){
    filterInput.value='';
    tbody.querySelectorAll('tr').forEach(r=>r.style.display='');
  }

  // popup show
  function showPopup(jobName, action='rilancia'){
    currentJob = jobName;
    popupTitle.textContent = `▶ ${action} build?`;
    popupMsg.textContent = `Vuoi ${action} il job "${jobName}"?`;
    overlay.style.display='flex';
  }

  function hidePopup(){
    overlay.style.display='none';
    currentJob=null;
  }

  // rilancio (simulato)
  function relaunchJob(jobName){
    showPopup(jobName, 'rilancia');
    // conferma -> esegue
    confirmBtn.onclick = function(){
      hidePopup();
      // cambia stato a pending per simulare
      const rows = tbody.querySelectorAll('tr');
      for(let tr of rows){
        const td = tr.querySelector('.job-name');
        if(td && td.textContent.trim() === jobName){
          const statusTd = tr.querySelector('td:nth-child(2)');
          if(statusTd){
            statusTd.innerHTML = badge('pending');
            // durata pending
            const durataTd = tr.querySelector('td:nth-child(4)');
            if(durataTd) durataTd.textContent = '--';
          }
          break;
        }
      }
      updateHero();
      applyFilter();
    };
    cancelBtn.onclick = hidePopup;
  }

  // eventi: pulsanti rilancia (delegation)
  tbody.addEventListener('click', e=>{
    const btn = e.target.closest('.action-btn');
    if(!btn) return;
    const job = btn.getAttribute('data-job');
    if(job) relaunchJob(job);
  });

  // aggiungi job
  addBtn.addEventListener('click', ()=>{
    const name = newName.value.trim();
    const status = newStatus.value.trim();
    if(!name || !status){
      alert('compila nome e stato');
      return;
    }
    addRow(name, status);
    newName.value='';
    newStatus.value='';
  });

  // filtro
  filterInput.addEventListener('input', applyFilter);
  resetBtn.addEventListener('click', resetFilter);

  // popup overlay chiudi con click fuori?
  overlay.addEventListener('click', e=>{
    if(e.target === overlay) hidePopup();
  });

  // inizializza hero
  updateHero();
  // eventuali righe aggiuntive? già presenti

})();
</script>
