<script>
  (function(){
    // ---------- data ----------
    let jobs = [
      { id: 'j1', name: 'frontend', status: 'ok', branch: 'main' },
      { id: 'j2', name: 'api-gateway', status: 'fail', branch: 'develop' },
      { id: 'j3', name: 'worker', status: 'pending', branch: 'feat/async' },
      { id: 'j4', name: 'mobile-build', status: 'ok', branch: 'release/1.2' },
    ];
    let nextId = 5;
    let currentFilter = '';

    // DOM refs
    const tbody = document.getElementById('tableBody');
    const heroStatus = document.getElementById('heroStatus');
    const filterInput = document.getElementById('filterInput');
    const resetBtn = document.getElementById('resetFilterBtn');
    const newName = document.getElementById('newJobName');
    const newBranch = document.getElementById('newJobBranch');
    const addBtn = document.getElementById('addJobBtn');
    const overlay = document.getElementById('confirmOverlay');
    const popupMsg = document.getElementById('popupMsg');
    const confirmBtn = document.getElementById('confirmBtn');
    const cancelBtn = document.getElementById('cancelBtn');

    // pending rebuild target
    let pendingRebuildId = null;

    // ---------- render ----------
    function render(){
      const filtered = jobs.filter(j => {
        if(!currentFilter) return true;
        return j.name.toLowerCase().includes(currentFilter.toLowerCase());
      });
      // build rows
      let html = '';
      filtered.forEach(j => {
        const statusClass = j.status === 'ok' ? 'badge ok' : (j.status === 'fail' ? 'badge fail' : 'badge pending');
        const statusLabel = j.status === 'ok' ? '✅ ok' : (j.status === 'fail' ? '❌ fail' : '⏳ pending');
        html += `<tr>
          <td class="job-name">${escapeHtml(j.name)}</td>
          <td><span class="${statusClass}">${statusLabel}</span></td>
          <td>${escapeHtml(j.branch)}</td>
          <td><button class="action-btn" data-rebuild-id="${j.id}">🔄 rebuild</button></td>
        </tr>`;
      });
      tbody.innerHTML = html;

      // attach rebuild events (after render)
      document.querySelectorAll('[data-rebuild-id]').forEach(btn => {
        btn.addEventListener('click', function(e){
          const id = this.getAttribute('data-rebuild-id');
          const job = jobs.find(j => j.id === id);
          if(!job) return;
          pendingRebuildId = id;
          popupMsg.textContent = `Restart build for "${job.name}" (${job.branch})?`;
          overlay.classList.add('show');
        });
      });

      // update hero status
      const okCount = jobs.filter(j => j.status === 'ok').length;
      heroStatus.textContent = `● ${okCount} OK / ${jobs.length} total`;
    }

    // helper escape
    function escapeHtml(str){
      const div = document.createElement('div');
      div.textContent = str;
      return div.innerHTML;
    }

    // ---------- actions ----------
    function addJob(){
      const name = newName.value.trim();
      const branch = newBranch.value.trim();
      if(!name || !branch) return;
      jobs.push({ id: 'j' + nextId++, name, status: 'pending', branch });
      newName.value = '';
      newBranch.value = '';
      render();
    }

    function confirmRebuild(){
      if(!pendingRebuildId) return;
      const idx = jobs.findIndex(j => j.id === pendingRebuildId);
      if(idx !== -1){
        // toggle status (simulate rebuild -> ok)
        jobs[idx].status = 'ok';
      }
      pendingRebuildId = null;
      overlay.classList.remove('show');
      render();
    }

    function cancelRebuild(){
      pendingRebuildId = null;
      overlay.classList.remove('show');
    }

    function applyFilter(){
      currentFilter = filterInput.value.trim();
      render();
    }

    function resetFilter(){
      filterInput.value = '';
      currentFilter = '';
      render();
    }

    // ---------- event binding ----------
    addBtn.addEventListener('click', addJob);
    confirmBtn.addEventListener('click', confirmRebuild);
    cancelBtn.addEventListener('click', cancelRebuild);
    filterInput.addEventListener('input', applyFilter);
    resetBtn.addEventListener('click', resetFilter);

    // close overlay on outside click (optional)
    overlay.addEventListener('click', function(e){
      if(e.target === overlay) cancelRebuild();
    });

    // initial render
    render();
  })();
</script>
