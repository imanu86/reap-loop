(function() {

    { id: 5, nome: "e2e-test", stato: "running", commit: "n4o5p6" },
  ];

  // elementi DOM
  const tabella = document.getElementById('tabella-job');
  const filtroInput = document.getElementById('filtro-input');
  const resetBtn = document.getElementById('reset-filtro');
  const aggiungiBtn = document.getElementById('aggiungi-job');
  const popup = document.getElementById('popup-conferma');
  const popupJobName = document.getElementById('popup-job-name');
  const siBtn = document.getElementById('popup-conferma-si');
  const noBtn = document.getElementById('popup-conferma-no');

  // stato popup
  let pendingJobId = null;

  // helper: aggiorna hero badge
  function aggiornaHero() {
    const ok = jobs.filter(j => j.stato === 'ok').length;
    const running = jobs.filter(j => j.stato === 'running').length;
    const fail = jobs.filter(j => j.stato === 'fail').length;
    document.getElementById('hero-ok').textContent = ✅ ${ok} OK;
    document.getElementById('hero-run').textContent = ⏳ ${running} in corso;
    document.getElementById('hero-fail').textContent = ❌ ${fail} fallite;
  }

  // render tabella con filtro
  function render(filtro = '') {
    const jobsFiltrati = jobs.filter(j =>
        j.nome.toLowerCase().includes(filtro.toLowerCase() || '')
        || j.stato.toLowerCase().includes(filtro.toLowerCase())
        || j.commit.toLowerCase().includes(filtro.toLowerCase())
    );
    tabella.innerHTML = jobsFiltrati.map(j =>
      <tr class="job-row" data-id="${j.id}">
        <td>${j.nome}</td>
        <td><span class="badge ${j.stato === 'ok' ? 'badge--ok' : 'badge--warn' : 'badge--fail'}">${j.stato}</span></td>
        <td><code>${j.commit}</code></td>
        <td class="azioni">
        <button class="btn btn--small rilancio" data-id="${j.id}">⟳ rilancia</button>
        </td>
    </tr>
    `);
    // attacca eventi rilancio
    document.querySelectorAll('.rilancio').forEach(b => {
      b.addEventListener('click', function(e) {
        const id = this.dataset.id;
        const job = jobs.find(j => j.id === id);
        if (job) {
          popupJobName.textContent = job.nome;
          pendingJobId = id;
          popup.classList.remove('hidden');
        }
      });
    // nessun bisogno di altri eventi
  }

  // rilancio confermato
  function rilanciaJob(id) {
    const job = jobs.find(j => j.id === id);
    if (job) {
      job.stato = 'running'; // simula rilancio
      aggiornaHero();
      render(filtroInput.value);
    }
    // chiudi popup
    popup.classList.add('hidden');
  }

  // aggiungi job
  function aggiungiJob() {
    const nome = document.getElementById('job-nome').value.trim();
    const commit = document.getElementById('job-commit').value.trim();
    if (nome && commit) {
      jobs.push({ id: Date.now(), nome, commit, stato: 'running' });
      aggiornaHero();
      render(filtroInput.value);
    // pulisci input
    document.getElementById('job-nome').value = '';
    document.getElementById('job-commit').value = '';
    // focus
    document.getElementById('job-nome').focus();
  }

  // filtro live
  filtroInput.addEventListener('input', function() {
    render(this.value);
  }

  resetBtn.addEventListener('click', function() {
    filtroInput.value = '';
    render('');
    filtroInput.focus();
  }

  // popup: si
  siBtn.addEventListener('click', function() {
    if (pendingJobId) {
      rilanciaJob(pendingJobId);
    // nessun pending
    }
  // popup: no
  noBtn.addEventListener('click', function() {
    popup.classList.add('hidden');
    pendingJobId = null;
  }

  // aggiungi job
  aggiungiBtn.addEventListener('click', aggiungiJob);

  // inizializza
  aggiornaHero();
  render('');
  // focus filtro
  filtroInput.focus();

  // chiusura popup cliccando overlay (opzionale)

    if (e.target === this) {
    // solo overlay
