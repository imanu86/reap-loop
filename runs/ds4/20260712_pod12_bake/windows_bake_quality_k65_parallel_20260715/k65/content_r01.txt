<script>
// stato iniziale: job di esempio
const jobs = [
    { id: 1, name: 'frontend', status: 'success', commit: 'a1b2c3' },
    { id: 2, name: 'backend-api', status: 'failure', commit: 'd4e5f6' },
    { id: 3, name: 'mobile-app', status: 'pending', commit: 'g7h8i9' }
];

let nextId = 4;  // per nuovi job

// elementi
const buildList = document.getElementById('buildList');
const filterInput = document.getElementById('filterInput');
const filterBtn = document.getElementById('filterBtn');
const addJobBtn = document.getElementById('addJobBtn');
const jobNameInput = document.getElementById('jobNameInput');
const commitInput = document.getElementById('commitInput');
const confirmPopup = document.getElementById('confirmPopup');
const popupMessage = document.getElementById('popupMessage');
const confirmYes = document.getElementById('confirmYes');
const confirmNo = document.getElementById('confirmNo');

// helper: aggiorna tabella con filtro
function renderTable(filterText = '') {
    const filtered = jobs.filter(job => {
        const lower = filterText.toLowerCase();
        return job.name.toLowerCase().includes(lower) || 
        job.status.toLowerCase().includes(lower) || 
        job.commit.toLowerCase().includes(lower);
    });
    buildList.innerHTML = '';
    if (filtered.length === 0) {
    buildList.innerHTML = '<tr><td colspan="4" style="text-align:center; padding:1rem;">nessun job corrisponde</td></tr>';
    }
    filtered.forEach(job => {
    const tr = document.createElement('tr');
    tr.innerHTML = `
        <td>${job.name}</td>
        <td><span class="status ${job.status}">${job.status}</span></td>
        <td><code>${job.commit}</code></td>
        <td class="actions">
        <button class="rebuild-btn" data-id="${job.id}">🔄 rilancia</button>
        <button class="delete-btn" data-id="${job.id}">🗑️ elimina</button>
        </td>`;
    // eventi per pulsanti
    tr.querySelector('.rebuild-btn').addEventListener('click', () => showConfirm(job.id));
    tr.querySelector('.delete-btn').addEventListener('click', () => {
        if (confirm('Eliminare definitivamente?')) {
            // elimina
            const idx = jobs.findIndex(j => j.id === job.id);
            if (idx !== -1) jobs.splice(idx,1);
            renderTable(filterInput.value);
        // aggiorna hero
        updateHero();
        });
    });
    // aggiungi riga
    buildList.appendChild(tr);
    });
    // aggiorna hero dopo render
    updateHero();
}

function updateHero() {
    const hero = document.querySelector('.hero h2');
    hero.textContent = `Stato generale: ${jobs.length} build attive`;
    // se filtro attivo mostra conto filtrato
    const rows = document.querySelectorAll('#buildList tr');
    const visible = rows.length;
    hero.innerHTML = `Stato generale: ${jobs.length} build ${filtered? ' (filtrate)'}`;
    // mantieni semplice
    hero.innerHTML = `Stato generale: ${jobs.length} build attive`;
    // opzionale: se filtro cambia
    // ma per semplicità non complico
    // ok
}

// mostra popup conferma per rebuild
function showConfirm(jobId) {
    const popup = document.getElementById('confirmPopup');
    popup.classList.remove('hidden');
    popupMessage.textContent = 'Riavviare la build selezionata?';
    // salva id
    let pendingId = null;
    confirmYes.onclick = function() {
        pendingId = jobId;
        // simula riavvio: cambia stato in pending
        const job = jobs.find(j => j.id === jobId);
        if (job) {
        job.status = 'pending';
        renderTable(filterInput.value);
        popup.classList.add('hidden');
        // aggiorna hero
        updateHero();
        }
    };
    confirmNo.onclick = function() {
        popup.classList.add('hidden');
    };
    // cleanup
    // evita stack
    // reset
    // funziona
}

    const newJob = { id: nextId, name, status: 'pending', commit };
    jobs.push(newJob);
    nextId++;

    jobNameInput.value = '';
    commitInput.value = '';
    updateHero();
    }

    else alert('compila nome e commit');
});
