<script>
// --- dati e stato ---
let jobs = [
  { name:'web-app', build:'#382', status:'ok' },
  { name:'api-gateway', build:'#179', status:'fail' },
  { name:'db-migrate', build:'#210', status:'pending' },
  { name:'auth-service', build:'#401', status:'ok' },
  { name:'payment-worker', build:'#98', status:'fail' },
  { name:'frontend-lint', build:'#556', status:'ok' }
];

let currentFilter = '';
let pendingRebuildJob = null; // per popup

// helpers
function statusClass(s) {
  if(s==='ok') return 'job-ok';
  if(s==='fail') return 'job-fail';
  return 'job-pending';
}
function statusLabel(s) {
  if(s==='ok') return '✅ successo';
  if(s==='fail') return '❌ fallito';
  return '⏳ in sospeso';
}

// aggiorna hero e tabella
function render() {
  const tbody = document.getElementById('tableBody');
  // filtro
  const f = currentFilter.toLowerCase();
  const filtered = jobs.filter(j => {
    if(!f) return true;
    const haystack = j.name.toLowerCase() + ' ' + j.status.toLowerCase();
    return haystack.includes(f);
  });

  let html = '';
  filtered.forEach(j => {
    const cls = statusClass(j.status);
    const label = statusLabel(j.status);
    html += `<tr><td>${j.name}</td><td>${j.build}</td><td class="${cls}">${label}</td><td class="actions"><button class="rebuild" data-job="${j.name}">🔄 rilancia</button></td></tr>`;
  });
  tbody.innerHTML = html;

  // hero status
  const total = jobs.length;
  const ok = jobs.filter(j=>j.status==='ok').length;
  const fail = jobs.filter(j=>j.status==='fail').length;
  const pend = jobs.filter(j=>j.status==='pending').length;
  document.getElementById('heroStatus').textContent = ok>fail ? '✅ '+ok+' OK' : (fail>0?'⚠️ alcuni falliti':'⏳ misto');
  document.getElementById('heroStatus').className = 'status' + (fail>0?' bad':'');
  // ultimo build (primo della lista)
  const lastBuild = jobs.length>0 ? jobs[0].build : '#---';
  document.getElementById('heroLast').textContent = lastBuild;

  // se nessun job filtrato, mostra riga vuota
  if(filtered.length===0) {
    tbody.innerHTML = `<tr><td colspan="4" style="text-align:center;padding:1rem;color:#6b7280">nessun job corrisponde al filtro</td></tr>`;
  }
}

// popup
function showPopup(msg, jobName) {
  document.getElementById('popupMsg').textContent = msg;
  pendingRebuildJob = jobName;
  document.getElementById('confirmPopup').classList.remove('hidden');
}
function hidePopup() {
  document.getElementById('confirmPopup').classList.add('hidden');
  pendingRebuildJob = null;
}

// rilancia (conferma)
function requestRebuild(jobName) {
  showPopup(`Rilanciare build per "${jobName}"?`, jobName);
}

// popup confirm -> esegue "rilancia" (simula cambio stato)
document.getElementById('popupConfirm').addEventListener('click', function(){
  if(pendingRebuildJob) {
    const idx = jobs.findIndex(j => j.name === pendingRebuildJob);
    if(idx !== -1) {
      // simula: stato -> pending, build incrementato
      jobs[idx].status = 'pending';
      const oldNum = parseInt(jobs[idx].build.replace('#','')) || 0;
      jobs[idx].build = '#' + (oldNum+1);
    }
    // eventuale riordino (sposta in alto?) no, solo update
    render();
    hidePopup();
  }
});
document.getElementById('popupCancel').addEventListener('click', hidePopup);
// click overlay chiude? no, solo pulsanti

// filtro
document.getElementById('filterInput').addEventListener('input', function(e){
  currentFilter = e.target.value;
  render();
});
document.getElementById('resetFilter').addEventListener('click', function(){
  document.getElementById('filterInput').value = '';
  currentFilter = '';
  render();
});

// aggiungi job
document.getElementById('addJobBtn').addEventListener('click', function(){
  const nameInput = document.getElementById('newJobName');
  const buildInput = document.getElementById('newJobBuild');
  const name = nameInput.value.trim();
  const build = buildInput.value.trim();
  if(!name || !build) {
    alert('compila entrambi i campi');
    return;
  }
  // evita duplicati (opzionale)
  if(jobs.some(j=>j.name===name)) {
    alert('job già esistente');
    return;
  }
  jobs.push({ name, build, status:'pending' });
  nameInput.value = '';
  buildInput.value = '';
  render();
});

// delegazione eventi per pulsanti rebuild (dinamici)
document.getElementById('jobTable').addEventListener('click', function(e){
  const btn = e.target.closest('.rebuild');
  if(!btn) return;
  const jobName = btn.getAttribute('data-job');
  if(jobName) requestRebuild(jobName);
});

// inizializza
render();
</script>
