<script>
(function(){
  // stato interno
  let jobs = [
    { name:'pipeline-auth', status:'success', commit:'2e3a1f' },
    { name:'ui-builder', status:'pending', commit:'bc7d4e' },
    { name:'api-gateway', status:'fail', commit:'9a1b2c' },
    { name:'db-migrate', status:'success', commit:'1f2d3e' }
  ];
  let filterText = '';
  let pendingRebuildJob = null;

  const tbody = document.getElementById('tableBody');
  const heroStatus = document.getElementById('heroStatus');
  const heroBadge = document.getElementById('heroBadge');
  const filterInput = document.getElementById('filterInput');
  const resetBtn = document.getElementById('resetFilterBtn');
  const addJobBtn = document.getElementById('addJobBtn');
  const newJobName = document.getElementById('newJobName');
  const newJobCommit = document.getElementById('newJobCommit');
  const confirmPopup = document.getElementById('confirmPopup');
  const popupJobName = document.getElementById('popupJobName');
  const popupCancel = document.getElementById('popupCancel');
  const popupConfirm = document.getElementById('popupConfirm');

  // helpers
  function statusIcon(s){
    if(s==='success') return '✅ success';
    if(s==='pending') return '⏳ pending';
    if(s==='fail') return '❌ fail';
    return '❓ '+s;
  }

  function updateHero(){
    const total = jobs.length;
    const success = jobs.filter(j=>j.status==='success').length;
    const pending = jobs.filter(j=>j.status==='pending').length;
    const fail = jobs.filter(j=>j.status==='fail').length;
    let msg = `✅ ${success} ok · ⏳ ${pending} in coda`;
    if(fail>0) msg += ` · ❌ ${fail} fail`;
    heroStatus.textContent = msg;
    heroBadge.textContent = total+' job';
  }

  function renderTable(){
    const filtered = jobs.filter(j=>{
      if(!filterText) return true;
      const f = filterText.toLowerCase();
      return j.name.toLowerCase().includes(f) || j.status.toLowerCase().includes(f);
    });
    if(filtered.length===0){
      tbody.innerHTML = `<tr><td colspan="4" style="text-align:center;padding:1.2rem;color:#64748b">nessun job corrisponde</td></tr>`;
      return;
    }
    let html = '';
    filtered.forEach(j=>{
      const icon = statusIcon(j.status);
      html += `<tr><td>${j.name}</td><td>${icon}</td><td>${j.commit}</td><td class="action"><button class="btn rebuild-trigger" data-job="${j.name}">🔄 rilancia</button></td></tr>`;
    });
    tbody.innerHTML = html;
    // assegna listener ai bottoni rebuild (anche dopo filtro)
    document.querySelectorAll('.rebuild-trigger').forEach(btn=>{
      btn.addEventListener('click', function(e){
        const job = this.dataset.job;
        pendingRebuildJob = job;
        popupJobName.textContent = `job: ${job}`;
        confirmPopup.classList.remove('hidden');
      });
    });
  }

  function rebuildJob(jobName){
    const idx = jobs.findIndex(j=>j.name===jobName);
    if(idx===-1) return;
    jobs[idx].status = 'pending';
    jobs[idx].commit = (Math.random()*16777215).toString(16).slice(0,6);
    renderTable();
    updateHero();
  }

  function addJob(name, commit){
    if(!name.trim()) return;
    const exists = jobs.some(j=>j.name===name.trim());
    if(exists) return;
    jobs.push({ name:name.trim(), status:'pending', commit:commit.trim() || '000000' });
    renderTable();
    updateHero();
  }

  function applyFilter(text){
    filterText = text;
    renderTable();
  }

  // eventi
  filterInput.addEventListener('input', function(){
    applyFilter(this.value);
  });
  resetBtn.addEventListener('click', function(){
    filterInput.value = '';
    filterText = '';
    renderTable();
  });

  addJobBtn.addEventListener('click', function(){
    const name = newJobName.value.trim();
    const commit = newJobCommit.value.trim();
    if(!name){ alert('inserire nome job'); return; }
    addJob(name, commit);
    newJobName.value = '';
    newJobCommit.value = '';
  });

  // popup
  popupCancel.addEventListener('click', function(){
    confirmPopup.classList.add('hidden');
    pendingRebuildJob = null;
  });
  popupConfirm.addEventListener('click', function(){
    if(pendingRebuildJob){
      rebuildJob(pendingRebuildJob);
    }
    confirmPopup.classList.add('hidden');
    pendingRebuildJob = null;
  });
  // click overlay per chiudere (opzionale)
  confirmPopup.addEventListener('click', function(e){
    if(e.target===confirmPopup){
      confirmPopup.classList.add('hidden');
      pendingRebuildJob = null;
    }
  });

  // inizializza
  renderTable();
  updateHero();
})();
</script>
