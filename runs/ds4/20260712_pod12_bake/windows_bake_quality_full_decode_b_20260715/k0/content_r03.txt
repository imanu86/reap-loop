<script>
// ----- in-memory data -----
let builds = [
  { job: 'frontend',   buildId: '#b7d3', status: 'success' },
  { job: 'backend-api', buildId: '#c21a', status: 'running' },
  { job: 'mobile-app',  buildId: '#e5f0', status: 'failed' },
  { job: 'docs',        buildId: '#a1b2', status: 'pending' }
];

let filteredBuilds = [...builds];

// ----- DOM refs -----
const tbody = document.getElementById('tableBody');
const filterInput = document.getElementById('filterInput');
const resetFilterBtn = document.getElementById('resetFilter');
const heroStatus = document.getElementById('heroStatus');
const newJobName = document.getElementById('newJobName');
const newBuildId = document.getElementById('newBuildId');
const addJobBtn = document.getElementById('addJobBtn');

// modal
const modalOverlay = document.getElementById('modalOverlay');
const modalMessage = document.getElementById('modalMessage');
const modalJobNameSpan = document.getElementById('modalJobName');
const modalConfirm = document.getElementById('modalConfirm');
const modalCancel = document.getElementById('modalCancel');

// ----- helpers -----
function updateHero() {
  const running = builds.filter(b => b.status === 'running').length;
  heroStatus.textContent = `✅ ${running} running`;
}

function renderTable(list) {
  if (!list) list = filteredBuilds;
  tbody.innerHTML = '';
  list.forEach(b => {
    const row = document.createElement('tr');
    const statusBadge = b.status === 'running' ? '🏃 running' : b.status === 'success' ? '✅ success' : b.status === 'failed' ? '❌ failed' : '⏳ pending';
    row.innerHTML = `
      <td><strong>${b.job}</strong></td>
      <td><span class="build-id">${b.buildId}</span></td>
      <td>${statusBadge}</td>
      <td><button class="btn outline rebuild-btn" data-job="${b.job}" data-build="${b.buildId}">⟳ rebuild</button></td>
    `;
    tbody.appendChild(row);
  });
  // attach rebuild listeners to each button
  document.querySelectorAll('.rebuild-btn').forEach(btn => {
    btn.addEventListener('click', (e) => {
      const job = btn.dataset.job;
      const buildId = btn.dataset.build;
      showModal(job, buildId);
    });
  });
}

// ----- filter -----
function applyFilter() {
  const q = filterInput.value.trim().toLowerCase();
  if (!q) {
    filteredBuilds = [...builds];
  } else {
    filteredBuilds = builds.filter(b => 
      b.job.toLowerCase().includes(q) || b.buildId.toLowerCase().includes(q)
    );
  }
  renderTable(filteredBuilds);
}

filterInput.addEventListener('input', applyFilter);
resetFilterBtn.addEventListener('click', () => {
  filterInput.value = '';
  applyFilter();
});

// ----- add job -----
addJobBtn.addEventListener('click', () => {
  const jobName = newJobName.value.trim();
  const buildId = newBuildId.value.trim();
  if (!jobName || !buildId) {
    alert('Please fill both job name and build id.');
    return;
  }
  // avoid duplicate build id? no check for simplicity
  builds.push({ job: jobName, buildId: buildId, status: 'pending' });
  newJobName.value = '';
  newBuildId.value = '';
  applyFilter();  // re-render with current filter
  updateHero();
});

// ----- modal logic -----
let pendingRebuildJob = null;
let pendingRebuildBuild = null;

function showModal(job, buildId) {
  pendingRebuildJob = job;
  pendingRebuildBuild = buildId;
  modalJobNameSpan.textContent = job;
  modalMessage.innerHTML = `Restart build for job <strong>${job}</strong> (${buildId})?`;
  modalOverlay.style.display = 'flex';
}

function hideModal() {
  modalOverlay.style.display = 'none';
  pendingRebuildJob = null;
  pendingRebuildBuild = null;
}

modalCancel.addEventListener('click', hideModal);
modalConfirm.addEventListener('click', () => {
  if (!pendingRebuildJob) return;
  // find build and set status to 'running' (simulate rebuild)
  const target = builds.find(b => b.job === pendingRebuildJob && b.buildId === pendingRebuildBuild);
  if (target) {
    target.status = 'running';
  }
  // reapply filter and update hero
  applyFilter();
  updateHero();
  hideModal();
});

// close modal on overlay click (background)
modalOverlay.addEventListener('click', (e) => {
  if (e.target === modalOverlay) hideModal();
});

// ----- initial render -----
renderTable(filteredBuilds);
updateHero();

// also update hero on any status change (rebuild already does)
</script>
