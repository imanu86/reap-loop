  (function() {
    // stato interno
    let jobs = [
      { name: 'frontend-build', status: 'pass', duration: '1m 12s' },
      { name: 'api-tests', status: 'fail', duration: '0m 48s' },
      { name: 'deploy-staging', status: 'pending', duration: '—' }
    ];

    const tbody = document.getElementById('tbodyJobs');
    const filterInput = document.getElementById('filterInput');
    const relanceBtn = document.getElementById('relanceBtn');
    const addJobBtn = document.getElementById('addJobBtn');
    const newJobName = document.getElementById('newJobName');
    const heroBadge = document.getElementById('heroStatusBadge');

    // popup
    const popup = document.getElementById('confirmPopup');
    const popupMsg = document.getElementById('popupMessage');
    const popupConfirm = document.getElementById('popupConfirm');
    const popupCancel = document.getElementById('popupCancel');
    let pendingJobName = null;  // job da rilanciare dopo conferma

    // helper aggiornamento tabella + hero
    function renderTable(filterText = '') {
      const filtered = jobs.filter(j => {
        if (!filterText) return true;
        const lower = filterText.toLowerCase();
        return j.name.toLowerCase().includes(lower) || j.status.toLowerCase().includes(lower);
      });

      let html = '';
      filtered.forEach(j => {
        let indicatorClass = 'pending';
        if (j.status === 'pass') indicatorClass = 'pass';
        else if (j.status === 'fail') indicatorClass = 'fail';
        else indicatorClass = 'pending';

        html += `<tr><td>🔨 ${j.name}</td><td><span class="status-indicator ${indicatorClass}"></span>${j.status}</td><td>${j.duration}</td><td><button class="action-btn" data-job="${j.name}">▶ rilancia</button></td></tr>`;
      });
      tbody.innerHTML = html;

      // aggancia event ai bottoni "rilancia" (action-btn)
      document.querySelectorAll('.action-btn').forEach(btn => {
        btn.addEventListener('click', function(e) {
          const jobName = this.dataset.job;
          pendingJobName = jobName;
          popupMsg.textContent = `Rilanciare il job "${jobName}"?`;
          popup.style.display = 'flex';
        });
      });

      // aggiorna hero badge
      const total = jobs.length;
      const passCount = jobs.filter(j => j.status === 'pass').length;
      const failCount = jobs.filter(j => j.status === 'fail').length;
      const pendingCount = jobs.filter(j => j.status === 'pending').length;
      let badgeClass = 'ok';
      let label = `✅ ${passCount} OK`;
      if (failCount > 0 && passCount === 0) { badgeClass = 'err'; label = `❌ ${failCount} fail`; }
      else if (failCount > 0) { badgeClass = 'warn'; label = `⚠️ ${passCount} OK, ${failCount} fail`; }
      else if (pendingCount > 0 && passCount === 0) { badgeClass = 'warn'; label = `⏳ ${pendingCount} pending`; }
      heroBadge.className = `badge ${badgeClass}`;
      heroBadge.textContent = label;
    }

    // filtro
    filterInput.addEventListener('input', function() {
      renderTable(this.value);
    });

    // rilancia selezionati (simula: cambia stato a pending per i job visibili)
    relanceBtn.addEventListener('click', function() {
      const filterText = filterInput.value;
      const visibleJobs = jobs.filter(j => {
        if (!filterText) return true;
        const lower = filterText.toLowerCase();
        return j.name.toLowerCase().includes(lower) || j.status.toLowerCase().includes(lower);
      });
      if (visibleJobs.length === 0) return;
      // chiedere conferma per tutti? usiamo popup generico
      pendingJobName = 'selected jobs';
      popupMsg.textContent = `Rilanciare ${visibleJobs.length} job selezionati?`;
      popup.style.display = 'flex';
      // override conferma per azione multipla
      popupConfirm._multi = true;
      popupConfirm._visibleJobs = visibleJobs.map(j => j.name);
      // usiamo gestione separata
    });

    // conferma popup (generico)
    function confirmAction() {
      if (pendingJobName === 'selected jobs') {
        // multi rilancia
        const names = popupConfirm._visibleJobs || [];
        names.forEach(jn => {
          const job = jobs.find(j => j.name === jn);
          if (job) { job.status = 'pending'; job.duration = '—'; }
        });
        popupConfirm._multi = false;
        popupConfirm._visibleJobs = [];
      } else if (pendingJobName) {
        // singolo job
        const job = jobs.find(j => j.name === pendingJobName);
        if (job) { job.status = 'pending'; job.duration = '—'; }
      }
      pendingJobName = null;
      popup.style.display = 'none';
      renderTable(filterInput.value);
    }

    popupConfirm.addEventListener('click', confirmAction);
    popupCancel.addEventListener('click', function() {
      popup.style.display = 'none';
      pendingJobName = null;
    });

    // aggiungi job
    addJobBtn.addEventListener('click', function() {
      const name = newJobName.value.trim();
      if (!name) return;
      // evita duplicati (opzionale)
      if (jobs.some(j => j.name === name)) {
        alert('
