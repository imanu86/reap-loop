<script>
// ---- dati iniziali (simulati) ----
let jobs = [
  { id: 'j1', name: 'backend-api', status: 'ok' },
  { id: 'j2', name: 'frontend-ui', status: 'fail' },
  { id: 'j3', name: 'mobile-app', status: 'pending' },
  { id: 'j4', name: 'data-pipeline', status: 'ok' },
  { id: 'j5', name: 'ci-tests', status: 'ok' },
];

let nextId = 6;

// popup state
let pendingRebuildId = null;

// DOM refs
const tbody = document.getElementById('tbodyJobs');
const filterInput = document.getElementById('filterInput');
const resetBtn = document.getElementById('resetFilterBtn');
const heroState = document.getElementById('heroState');
const newName = document.getElementById('newJobName');
const newStatus = document.getElementById('newJobStatus');
const addBtn = document.getElementById('addJobBtn');
const popupOverlay = document.getElementById('confirmPopup');
const popupMsg = document.getElementById('popupMessage');
const confirmBtn = document.getElementById('popupConfirmBtn');
const cancelBtn = document.getElementById('popupCancelBtn');

// ---- render ----
function render(filterText = '') {
  const f = filterText.toLowerCase().trim();
  const filtered = jobs.filter(j => {
    if (!f) return true;
    return j.name.toLowerCase().includes(f) || j.status.toLowerCase().includes(f);
  });

  let html = '';
  for (const j of filtered) {
    const badgeClass = j.status === 'ok' ? 'badge ok' : (j.status === 'fail' ? 'badge fail' : 'badge pending');
    const label = j.status === 'pending' ? 'pending' : j.status;
    html += `<tr class="job-row" data-id="${j.id}">
      <td>${j.name}</td>
      <td><span class="${badgeClass}">${label}</span></td>
      <td><button class="btn-small" data-rebuild-id="${j.id}">⟳ rilancia</button></td>
    </tr>`;
  }
  tbody.innerHTML = html;

  // aggancia eventi ai bottoni "rilancia"
  tbody.querySelectorAll('[data-rebuild-id]').forEach(btn => {
    btn.addEventListener('click', function(e) {
      const id = this.dataset.rebuildId;
      pendingRebuildId = id;
      const job = jobs.find(j => j.id === id);
      popupMsg.textContent = `rilanciare build "${job ? job.name : '?'}" ?`;
      popupOverlay.style.display = 'flex';
    });
  });

  // aggiorna stato hero
  const total = jobs.length;
  const ok = jobs.filter(j => j.status === 'ok').length;
  const fail = jobs.filter(j => j.status === 'fail').length;
  const pending = jobs.filter(j => j.status === 'pending').length;
  let heroText = `✅ ${ok} ok · ❌ ${fail} fail · ⏳ ${pending} pending`;
  if (total === 0) heroText = '📡 nessun job';
  heroState.textContent = heroText;
}

// ---- popup gestione ----
function closePopup() {
  popupOverlay.style.display = 'none';
  pendingRebuildId = null;
}

confirmBtn.addEventListener('click', function() {
  if (pendingRebuildId) {
    const job = jobs.find(j => j.id === pendingRebuildId);
    if (job) {
      // simula rilancio → pending
      job.status = 'pending';
      render(filterInput.value);
    }
  }
  closePopup();
});

cancelBtn.addEventListener('click', closePopup);
popupOverlay.addEventListener('click', function(e) {
  if (e.target === popupOverlay) closePopup();
});

// ---- filtro ----
filterInput.addEventListener('input', function() {
  render(this.value);
});
resetBtn.addEventListener('click', function() {
  filterInput.value = '';
  render('');
});

// ---- aggiungi job ----
addBtn.addEventListener('click', function() {
  const name = newName.value.trim();
  const statusRaw = newStatus.value.trim().toLowerCase();
  if (!name) return;
  const validStatus = ['ok', 'fail', 'pending'];
  const status = validStatus.includes(statusRaw) ? statusRaw : 'pending';
  const newJob = { id: 'j' + (nextId++), name, status };
  jobs.push(newJob);
  render(filterInput.value);
  newName.value = '';
  newStatus.value = '';
});

// ---- avvio ----
render('');
</script>
