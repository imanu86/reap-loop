<script>
  (function() {
    // ---------- DATI INIZIALI (simulazione build) ----------
    let jobs = [
      { id: 'j1', name: 'frontend-ci', status: 'success', lastRun: '2 min ago' },
      { id: 'j2', name: 'api-gateway', status: 'fail', lastRun: '15 min ago' },
      { id: 'j3', name: 'worker-email', status: 'running', lastRun: 'ora' },
      { id: 'j4', name: 'auth-service', status: 'success', lastRun: '1h ago' },
      { id: 'j5', name: 'cron-backup', status: 'fail', lastRun: '3h ago' },
    ];
    let nextId = 100;

    // riferimenti DOM
    const tbody = document.getElementById('tableBody');
    const filterInput = document.getElementById('filterInput');
    const heroStatus = document.getElementById('heroStatus');
    const restartBtn = document.getElementById('restartBtn');
    const addJobBtn = document.getElementById('addJobBtn');
    const addForm = document.getElementById('addForm');
    const newJobName = document.getElementById('newJobName');
    const newJobStatus = document.getElementById('newJobStatus');
    const confirmAddBtn = document.getElementById('confirmAddJob');

    // modale
    const modal = document.getElementById('confirmModal');
    const modalMsg = document.getElementById('modalMessage');
    const modalConfirm = document.getElementById('modalConfirm');
    const modalCancel = document.getElementById('modalCancel');
    let pendingCallback = null;  // funzione da eseguire dopo conferma

    // helper: mostra modale con messaggio e callback
    function showConfirm(message, onConfirm) {
      modalMsg.textContent = message;
      modal.style.display = 'flex';
      pendingCallback = onConfirm;
    }

    function hideModal() {
      modal.style.display = 'none';
      pendingCallback = null;
    }

    modalCancel.addEventListener('click', hideModal);
    modalConfirm.addEventListener('click', function() {
      if (pendingCallback) pendingCallback();
      hideModal();
    });

    // aggiornamento hero (stato generale: se almeno un fail => warning, se tutti success => stable, se running misto => running)
    function updateHero() {
      const hasFail = jobs.some(j => j.status === 'fail');
      const hasRunning = jobs.some(j => j.status === 'running');
      const allSuccess = jobs.every(j => j.status === 'success');
      let text = '✅ stable';
      let cls = 'status';
      if (hasFail) { text = '⚠️ failures'; cls += ' warning'; }
      else if (hasRunning) { text = '⏳ running'; cls += ''; } // default blue? ma usiamo running badge
      else if (allSuccess) { text = '✅ all green'; cls += ''; }
      heroStatus.textContent = text;
      heroStatus.className = cls;
    }

    // render tabella con filtro attivo
    function renderTable() {
      const filterVal = filterInput.value.toLowerCase();
      const filtered = jobs.filter(j => {
        if (!filterVal) return true;
        return j.name.toLowerCase().includes(filterVal) || j.status.toLowerCase().includes(filterVal);
      });

      let html = '';
      filtered.forEach(j => {
        let badgeClass = 'badge';
        if (j.status === 'success') badgeClass += ' success';
        else if (j.status === 'fail') badgeClass += ' fail';
        else if (j.status === 'running') badgeClass += ' running';
        else badgeClass += ''; // default

        html += `<tr>
          <td class="job-name">${j.name}</td>
          <td><span class="${badgeClass}">${j.status}</span></td>
          <td>${j.lastRun}</td>
          <td><button class="action-btn" data-id="${j.id}" data-action="restart">⟳ rilancia</button></td>
        </tr>`;
      });
      tbody.innerHTML = html;

      // assegna event ai bottoni "rilancia" (per singolo job)
      document.querySelectorAll('[data-action="restart"]').forEach(btn => {
        btn.addEventListener('click', function(e) {
          const id = this.dataset.id;
          const job = jobs.find(j => j.id === id);
          if (!job) return;
          // chiedere conferma per rilancio singolo
          showConfirm(`Rilanciare build "${job.name}" ?`, function() {
            job.status = 'running';
            job.lastRun = 'ora';
            renderTable();
            updateHero();
          });
        });
      });
    }

    // rilancio tutte le build fallite (conferma)
    restartBtn.addEventListener('click', function() {
      const failed = jobs.filter(j => j.status === 'fail');
      if (failed.length === 0) {
        alert('Nessuna build fallita da rilanciare.');
        return;
      }
      const names = failed.map(j => j.name).join(', ');
      showConfirm(`Rilanciare ${failed.length} build: ${names
