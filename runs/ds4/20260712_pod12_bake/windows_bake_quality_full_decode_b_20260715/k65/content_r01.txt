<script>
// ---------- DATA ----------
let jobs = [
  { id: 'j1', name: 'frontend-ci', status: 'ok' },
  { id: 'j2', name: 'api-build', status: 'fail' },
  { id: 'j3', name: 'worker-test', status: 'pending' },
  { id: 'j4', name: 'mobile-lint', status: 'ok' },
];
let nextId = 100;

// DOM refs
const tbody = document.getElementById('tableBody');
const filterInput = document.getElementById('filterInput');
const heroStatus = document.getElementById('heroStatus');
const jobCountSpan = document.getElementById('jobCount');
const addForm = document.getElementById('addForm');
const showAddBtn = document.getElementById('showAddBtn');
const closeAddBtn = document.getElementById('closeAddBtn');
const newJobName = document.getElementById('newJobName');
const newJobStatus = document.getElementById('newJobStatus');
const saveJobBtn = document.getElementById('saveJobBtn');
const modalOverlay = document.getElementById('confirmModal');
const modalMessage = document.getElementById('modalMessage');
const modalConfirmBtn = document.getElementById('modalConfirmBtn');
const modalCancelBtn = document.getElementById('modalCancelBtn');

// pending callback (rilancia)
let pendingRelaunchId = null;

// ---------- RENDER ----------
function renderTable() {
  const filterText = filterInput.value.toLowerCase();
  const filtered = jobs.filter(j => {
    const nameMatch = j.name.toLowerCase().includes(filterText);
    const statusMatch = j.status.toLowerCase().includes(filterText);
    return nameMatch || statusMatch;
  });

  let html = '';
  filtered.forEach(j => {
    const statusClass = j.status === 'ok' ? 'status-ok' : (j.status === 'fail' ? 'status-fail' : 'status-pending');
    const statusLabel = j.status === 'ok' ? 'successo' : (j.status === 'fail' ? 'fallito' : 'in sospeso');
    html += `<tr>
      <td><strong>${j.name}</strong></td>
      <td><span class="status-indicator ${statusClass}"></span>${statusLabel}</td>
      <td><button class="action-btn" data-id="${j.id}">🔄 rilancia</button></td>
    </tr>`;
  });
  tbody.innerHTML = html;

  // attach event per pulsanti rilancio
  tbody.querySelectorAll('.action-btn').forEach(btn => {
    btn.addEventListener('click', (e) => {
      const id = btn.getAttribute('data-id');
      const job = jobs.find(j => j.id === id);
      if (!job) return;
      pendingRelaunchId = id;
      modalMessage.textContent = `Rilanciare build "${job.name}" ?`;
      modalOverlay.style.display = 'flex';
    });
  });

  // aggiorna hero e count
  const okCount = jobs.filter(j => j.status === 'ok').length;
  const failCount = jobs.filter(j => j.status === 'fail').length;
  const pendingCount = jobs.filter(j => j.status === 'pending').length;
  let heroText = '';
  if (failCount > 0) heroText = '⚠️ attenzione (fallimenti)';
  else if (pendingCount > 0) heroText = '⏳ pending in corso';
  else heroText = '✅ operativo';
  heroStatus.textContent = heroText;
  jobCountSpan.textContent = jobs.length;
}

// ---------- MODAL ----------
modalConfirmBtn.addEventListener('click', () => {
  if (pendingRelaunchId) {
    const job = jobs.find(j => j.id === pendingRelaunchId);
    if (job) {
      // simula rilancia -> pending
      job.status = 'pending';
      renderTable();
    }
    pendingRelaunchId = null;
  }
  modalOverlay.style.display = 'none';
});
modalCancelBtn.addEventListener('click', () => {
  pendingRelaunchId = null;
  modalOverlay.style.display = 'none';
});
window.addEventListener('click', (e) => {
  if (e.target === modalOverlay) {
    pendingRelaunchId = null;
    modalOverlay.style.display = 'none';
  }
});

// ---------- FILTER ----------
filterInput.addEventListener('input', renderTable);

// ---------- ADD FORM TOGGLE ----------
showAddBtn.addEventListener('click', () => {
  addForm.style.display = 'flex';
});
closeAddBtn.addEventListener('click', () => {
  addForm.style.display = 'none';
  newJobName.value = '';
  newJobStatus.value = '';
});

saveJobBtn.addEventListener('click', () => {
  const name = newJobName.value.trim();
  const status = newJobStatus.value.trim().toLowerCase();
  if (!name) return alert('inserire nome job');
  const validStatuses = ['ok', 'fail', 'pending'];
  if (!validStatuses.includes(status)) return alert('stato deve essere ok, fail o pending');
  const newJob = { id: 'j' + nextId++, name, status };
  jobs.push(newJob);
  renderTable();
  addForm.style.display = 'none';
  newJobName.value = '';
  newJobStatus.value = '';
});

// ---------- INIT ----------
renderTable();
</script>
