                rows += `<tr data-name="${j.name}" data-status="${j.status}"><td class="job-name">${j.name}</td><td><span class="status-badge ${badgeClass}">${j.status}</span></td><td>${j.last}</td><td><button class="action-btn" data-action="rebuild">▶ rebuild</button></td></tr>`;

        function updateGlobalStatus() {
            const success = jobs.filter(j => j.status === 'success').length;
