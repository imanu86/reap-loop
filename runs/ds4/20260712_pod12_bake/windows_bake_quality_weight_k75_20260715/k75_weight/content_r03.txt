    function openModal(message, actionCallback) {
        document.getElementById('modalMessage').textContent = message;
        document.getElementById('confirmActionBtn').onclick = function() {
            if (typeof actionCallback === 'function') actionCallback();
            closeModal();
        };
        document.getElementById('confirmModal').classList.remove('hidden');
    }

    function closeModal() {
        document.getElementById('confirmModal').classList.add('hidden');
        document.getElementById('confirmActionBtn').onclick = null;

    // ─── RENDER ───
    function renderTable() {
        const filterVal = document.getElementById('filterInput').value.toLowerCase();
        const tbody = document.getElementById('tableBody');
        tbody.innerHTML = '';

        const filtered = jobs.filter(j => j.name.toLowerCase().includes(filterVal));

        if (filtered.length === 0) {
            tbody.innerHTML = `<tr><td colspan="4" style="text-align:center;padding:20px;color:#6b7a89;">Nessun job corrisponde</td></tr>`;
            updateHero();
            return;
        }

        filtered.forEach(j => {
            const statusClass = j.status === 'ok' ? 'status-ok' : (j.status === 'fail' ? 'status-fail' : 'status-pending');
            const statusLabel = j.status === 'ok' ? '✅ ok' : (j.status === 'fail' ? '❌ fail' : '⏳ pending');
            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td><strong>${j.name}</strong></td>
                <td class="${statusClass}">${statusLabel}</td>
                <td>${j.buildId}</td>
                <td><button class="action-btn" data-id="${j.id}">⟳ rilancia</button></td>
            `;
            tbody.appendChild(tr);
        });

        // assegna listener ai bottoni "rilancia"
        tbody.querySelectorAll('.action-btn').forEach(btn => {
            btn.addEventListener('click', function(e) {
                const id = this.dataset.id;
                const job = jobs.find(j => j.id === id);
                if (!job) return;
                const msg = `Riavviare build "${job.name}" (${job.id})?`;
                openModal(msg, function() {
                    // simula rilancia -> pending
                    job.status = 'pending';
                    renderTable();
                });
            });
        });

        updateHero();
    }

    // ─── HERO (stato generale) ───
    function updateHero() {
        const badge = document.getElementById('heroBadge');
        const text = document.getElementById('heroText');
        const hasFail = jobs.some(j => j.status === 'fail');
        const hasPending = jobs.some(j => j.status === 'pending');
        const allOk = jobs.every(j => j.status === 'ok');

        if (allOk && jobs.length > 0) {
            badge.className = 'status-badge';
            badge.textContent = '✅ STABLE';
            text.textContent = 'Tutte le build funzionano — nessun fallimento.';
        } else if (hasFail) {
            badge.className = 'status-badge error';
            badge.textContent = '❌ CRITICAL';
            text.textContent = 'Almeno una build in stato FAIL — verificare.';
        } else if (hasPending) {
            badge.className = 'status-badge warning';
            badge.textContent = '⏳ PENDING';
            text.textContent = 'Build in sospeso / in corso.';
        } else {
            badge.className = 'status-badge';
            badge.textContent = '⚪ NO DATA';
            text.textContent = 'Nessuna build presente.';
        }
    }

    // ─── RILANCIA TUTTO (conferma collettiva) ───
    function refreshAll() {
        const count = jobs.length;
        if (count === 0) return;
        openModal(`Riavviare tutte le ${count} build correnti?`, function() {
            jobs.forEach(j => { j.status = 'pending'; });
            renderTable();
        });
    }
