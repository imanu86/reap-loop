        (function() {
            // --- stato ---
            const tbody = document.getElementById('tbody');
            const filterInput = document.getElementById('filterInput');

            const confirmBtn = document.getElementById('popupConfirm');

            // dati interni per restart callback
            let pendingRestartJob = null;

            // helper: aggiorna badge hero (stato generale)
            function updateHeroStatus() {
                const rows = tbody.querySelectorAll('tr');
                let ok = 0, fail = 0, progress = 0;
                rows.forEach(row => {
                    const badge = row.querySelector('.status-badge-sm');
                    if (!badge) return;
                    const text = badge.textContent.trim().toLowerCase();
                    if (text === 'success') ok++;
                    else if (text === 'failed') fail++;
                    else if (text === 'building') progress++;
                });
                const heroBadge = document.querySelector('.hero .status-badge');
                let statusText = `● ${ok} OK · ${fail} FAIL`;
                if (progress > 0) statusText += ` · ${progress} building`;
                heroBadge.textContent = statusText;
            }

            // restart job (simulato) con popup
            function requestRestart(jobName) {
                pendingRestartJob = jobName;
                popupMsg.innerHTML = `restart build for <strong>${jobName}</strong>?`;
                overlay.style.display = 'flex';
            }

            // conferma restart
            function confirmRestart() {
                if (!pendingRestartJob) return;
                const job = pendingRestartJob;
                // simula: cambia stato a "building" per il job specifico
                const rows = tbody.querySelectorAll('tr');
                for (let row of rows) {
                    const nameCell = row.querySelector('.job-name');
                    if (nameCell && nameCell.textContent.trim() === job) {
                        const badge = row.querySelector('.status-badge-sm');
                        if (badge) {
                            badge.className = 'status-badge-sm status-progress';
                            badge.textContent = 'building';
                        }
                        break;
                    }
                }
                pendingRestartJob = null;
                overlay.style.display = 'none';
                updateHeroStatus();
            }

            // cancellazione popup
            function cancelPopup() {
                pendingRestartJob = null;
                overlay.style.display = 'none';
            }

                const target = e.target.closest('.restart-btn');
                if (!target) return;
                const job = target.getAttribute('data-job');
                if (job) requestRestart(job);
