        function updateHero() {
            const rows = tableBody.querySelectorAll('tr');
            let ok = 0, warn = 0, err = 0;
            rows.forEach(tr => {
                const status = (tr.dataset.status || '').toLowerCase();
                if (status === 'ok') ok++;
                else if (status === 'warn') warn++;
                else if (status === 'err') err++;
            });
            const total = ok + warn + err;
            let badgeClass = 'badge ok';
            let text = `✅ ${ok} ok`;
            if (warn > 0) { text += ` · ${warn} warn`; badgeClass = 'badge warn'; }
            if (err > 0) { text += ` · ${err} err`; badgeClass = 'badge err'; }
            if (warn === 0 && err === 0) badgeClass = 'badge ok';
            heroBadge.className = badgeClass;
            heroBadge.textContent = text;
        }

        function applyFilter() {
            const q = filterInput.value.toLowerCase().trim();
            const rows = tableBody.querySelectorAll('tr');
            rows.forEach(tr => {
                const job = (tr.dataset.job || '').toLowerCase();
                const branch = (tr.dataset.branch || '').toLowerCase();
                const status = (tr.dataset.status || '').toLowerCase();
                const text = job + ' ' + branch + ' ' + status;
                if (!q || text.includes(q)) {
                    tr.style.display = '';
                } else {
                    tr.style.display = 'none';
                }
            });
            updateHero();
        }

            // find row
