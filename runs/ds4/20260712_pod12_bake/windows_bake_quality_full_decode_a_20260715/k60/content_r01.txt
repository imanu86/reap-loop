<script>
// ---------- DATI INIZIALI ----------
let builds = [
  { id: 'b-101', job: 'frontend', commit: 'e2a1f', status: 'success' },
  { id: 'b-102', job: 'api-gateway', commit: 'c3d4e', status: 'fail' },
  { id: 'b-103', job: 'worker-queue', commit: 'f5g6h', status: 'pending' },
  { id: 'b-104', job: 'frontend', commit: 'a7b8c', status: 'success' },
  { id: 'b-105', job: 'auth-service', commit: 'd9e0f', status: 'success' },
];

let nextId = 106;
let currentFilter = '';

// elementi
const tbody = document.getElementById('tableBody');
const heroStatus = document.getElementById('heroStatus');
const filterInput = document.getElementById('filterInput');
const resetBtn = document.getElementById('resetFilterBtn');
const newJobName = document.getElementById('newJobName');
const newCommitId = document.getElementById('newCommitId');
const addBtn = document.getElementById('addJobBtn');
const popupOverlay = document.getElementById('confirmPopup');
const popupJobInfo = document.getElementById('popupJobInfo');
const popupConfirm = document.getElementById('popupConfirmBtn');
const popupCancel = document.getElementById('popupCancelBtn');

// popup state
let pendingRestartId = null;

// ---------- RENDER ----------
function renderTable() {
  const filter = currentFilter.toLowerCase();
  const filtered = builds.filter(b => {
    if (!filter) return true;
    return b.job.toLowerCase().includes(filter) ||
           b.commit.toLowerCase().includes(filter) ||
           b.status.toLowerCase().includes(filter);
  });

  let html = '';
  filtered.forEach(b => {
    const statusClass = b.status === 'success' ? 'success' : b.status === 'fail' ? 'fail' : 'pending';
    html += `<tr>
      <td><span class="job-id">${b.job}</span></td>
      <td><code>${b.commit}</code></td>
      <td><span class="status-badge ${statusClass}">${b.status}</span></td>
      <td><button class="action-btn" data-id="${b.id}">⟳ restart</button></td>
    </tr>`;
  });
  tbody.innerHTML = html;

  // aggancia eventi restart
  tbody.querySelectorAll('.action-btn').forEach(btn => {
    btn.addEventListener('click', (e) => {
      const id = e.target.dataset.id;
      const build = builds.find(b => b.id === id);
      if (!build) return;
      pendingRestartId = id;
      popupJobInfo.textContent = `job: ${build.job} · commit: ${build.commit}`;
      popupOverlay.style.display = 'flex';
    });
  });

  updateHeroStatus();
}

function updateHeroStatus() {
  const hasFail = builds.some(b => b.status === 'fail');
  const hasPending = builds.some(b => b.status === 'pending');
  let text;
  if (hasFail) text = '❌ failures detected';
  else if (hasPending) text = '⏳ pending builds';
  else text = '✅ all stable';
  heroStatus.textContent = text;
}

// ---------- FILTRO ----------
filterInput.addEventListener('input', () => {
  currentFilter = filterInput.value;
  renderTable();
});
resetBtn.addEventListener('click', () => {
  filterInput.value = '';
  currentFilter = '';
  renderTable();
});

// ---------- AGGIUNTA JOB ----------
addBtn.addEventListener('click', () => {
  const job = newJobName.value.trim();
  const commit = newCommitId.value.trim();
  if (!job || !commit) return alert('job and commit required');
  const newBuild = {
    id: `b-${nextId++}`,
    job: job,
    commit: commit,
    status: 'pending'
  };
  builds.push(newBuild);
  renderTable();
  newJobName.value = '';
  newCommitId.value = '';
});

// ---------- POPUP CONFERMA ----------
popupConfirm.addEventListener('click', () => {
  if (!pendingRestartId) return;
  const idx = builds.findIndex(b => b.id === pendingRestartId);
  if (idx !== -1) {
    builds[idx].status = 'pending';
    renderTable();
  }
  pendingRestartId = null;
  popupOverlay.style.display = 'none';
});

popupCancel.addEventListener('click', () => {
  pendingRestartId = null;
  popupOverlay.style.display = 'none';
});

// click overlay per chiudere (opzionale)
popupOverlay.addEventListener('click', (e) => {
  if (e.target === popupOverlay) {
    pendingRestartId = null;
    popupOverlay.style.display = 'none';
  }
});

// ---------- INIT ----------
renderTable();
</script>
