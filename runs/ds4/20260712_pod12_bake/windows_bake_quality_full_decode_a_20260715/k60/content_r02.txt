        .hero h1 { color: #f8fafc; font-size: 1.8rem; font-weight: 600; }

        .badge { display: inline-block; padding: 0.2rem 0.8rem; border-radius: 40px; font-size: 0.75rem; font-weight: 600; }
        .badge.success { background: #dcfce7; color: #166534; }
        .badge.fail { background: #fee2e2; color: #991b1b; }
        .badge.pending { background: #fef9c3; color: #92400e; }
        .actions { display: flex; gap: 0.6rem; }
        .actions button { background: transparent; border: 1px solid #d1d5db; padding: 0.3rem 0.9rem; border-radius: 40px; font-size: 0.75rem; font-weight: 500; cursor: pointer; transition: 0.2s; }
        .actions button:hover { background: #f1f5f9; border-color: #0f172a; }

    <!-- NAV -->

    <div class="card">
        <div class="filter">
            <input type="text" id="filterInput" placeholder="🔍 filtra per nome, stato…" oninput="applyFilter()">
            <button onclick="resetFilter()">Reset</button>
        </div>
        <table id="jobTable">
            <thead><tr><th>Job</th><th>Build</th><th>Stato</th><th>Azioni</th></tr></thead>
            <tbody id="tableBody">
                <!-- righe dinamiche -->
            </tbody>
        </table>

        <!-- modulo aggiunta job -->
        <div class="add-form">
            <input type="text" id="newJobName" placeholder="nome job (es: frontend)">
            <input type="text" id="newBuildId" placeholder="build ID (es: #A42)">
            <button onclick="addJob()">➕ Aggiungi</button>
        </div>
    </div>

    <!-- POPUP CONFERMA (rilancia) -->
    <div class="popup-overlay" id="confirmPopup">
        <div class="popup-box">
            <p id="popupMsg">Rilanciare la build selezionata?</p>
            <div class="flex-row">
                <button id="confirmYes" onclick="confirmAction()">Conferma</button>
                <button class="cancel" onclick="closePopup()">Annulla</button>

    <script>
        // ---------- DATI INIZIALI ----------
        let jobs = [
            { id: 'j1', name: 'frontend', build: '#B232', status: 'success' },
            { id: 'j2', name: 'api-gateway', build: '#C901', status: 'fail' },
            { id: 'j3', name: 'worker-queue', build: '#D113', status: 'pending' },
            { id: 'j4', name: 'auth-service', build: '#E445', status: 'success' },
        ];

        let nextId = 100;
        let pendingReloadId = null;   // per popup

        // ---------- RENDER ----------
        function renderTable(filterText = '') {
            const tbody = document.getElementById('tableBody');
            let html = '';
            const lowerFilter = filterText.toLowerCase();

            jobs.forEach(job => {
                const match = !filterText || 
                    job.name.toLowerCase().includes(lowerFilter) || 
                    job.build.toLowerCase().includes(lowerFilter) || 
                    job.status.toLowerCase().includes(lowerFilter);
                if (!match) return;

                const badgeClass = job.status === 'success' ? 'badge success' : 
                                   job.status === 'fail' ? 'badge fail' : 'badge pending';
                const statusLabel = job.status === 'success' ? 'success' : 
                                    job.status === 'fail' ? 'fail' : 'pending';

                html += `<tr>
                    <td><strong>${job.name}</strong></td>
                    <td>${job.build}</td>
                    <td><span class="${badgeClass}">${statusLabel}</span></td>
                    <td class="actions">
                        <button onclick="requestRerun('${job.id}')">⟳ rilancia</button>
                        <button onclick="removeJob('${job.id}')">🗑</button>
                    </td>
                </tr>`;
            });

            tbody.innerHTML = html;
            updateHeroState();
        }

        // ---------- AGGIORNA STATO GENERALE (hero) ----------
        function updateHeroState() {
            const allStatuses = jobs.map(j => j.status);
            let overall = '✅ STABLE';
            if (allStatuses.includes('fail')) overall = '🔴 FAILING';
            else if (allStatuses.includes('pending')) overall = '⏳ IN PROGRESS';
            document.getElementById('heroState').textContent = overall;
        }

        // ---------- FILTRO ----------
        function applyFilter() {
            const val = document.getElementById('filterInput').value;
            renderTable(val);
        }

        function resetFilter() {
            document.getElementById('filterInput').value = '';
            renderTable('');
        }

        // ---------- AGGIUNGI JOB ----------
        function addJob() {
            const nameInput = document.getElementById('newJobName');
            const buildInput = document.getElementById('newBuildId');
            const name = nameInput.value.trim();
            const build = buildInput.value.trim();
            if (!name || !build) return alert('compila entrambi i campi');

            const newJob = {
                id: 'j' + (nextId++),
                name: name,
                build: build,
                status: 'pending'   // default pending
            };
            jobs.push(newJob);
            nameInput.value = '';
            buildInput.value = '';
            renderTable(document.getElementById('filterInput').value);
        }

        // ---------- RILANCIA (popup) ----------
        function requestRerun(jobId) {
            pendingReloadId = jobId;
            const job = jobs.find(j => j.id === jobId);
            document.getElementById('popupMsg').textContent = `Rilanciare build ${job.build} (${job.name})?`;
            document.getElementById('confirmPopup').style.display = 'flex';
        }

        function confirmAction() {
            if (!pendingReloadId) return;
            const idx = jobs.findIndex(j => j.id === pendingReloadId);
            if (idx !== -1) {
                jobs[idx].status = 'pending';   // rilancia → pending
            }
            pendingReloadId = null;
            closePopup();
            renderTable(document.getElementById('filterInput').value);
        }

        function closePopup() {
            document.getElementById('confirmPopup').style.display = 'none';
            pendingReloadId = null;
        }

        // ---------- RIMUOVI JOB ----------
        function removeJob(jobId) {
            const idx = jobs.findIndex(j => j.id === jobId);
            if (idx !== -1) {
                jobs.splice(idx, 1);
                renderTable(document.getElementById('filterInput').value);
            }
        }

        // ---------- AVVIO ----------
        renderTable('');
    </script>
