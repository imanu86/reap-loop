<script>
//  Single‑file JavaScript, nessuna libreria esterna
// Dati iniziali
const jobs = [
{ id: 'job-1', name: 'backend-api', status: 'success', last: '14:23:11' },
{ id: 'job-2', name: 'frontend-ui', status: 'fail', last: '14:22:04' },
{ id: 'job-3', name: 'mobile-build', status: 'pending', last: '14:21:55' },
{ id: 'job-4', name: 'e2e-tests', status: 'success', last: '14:20:30' },
{ id: 'job-5', name: 'docs-gen', status: 'paused', last: '14:19:12' },
];

let currentFilter = '';
let pendingPopupAction = null; // per popup

// render tabella + badge hero
function renderTable() {
const tbody = document.getElementById('buildBody');
const filter = document.getElementById('filterInput').value.toLowerCase().trim();
const filtered = jobs.filter(j => {
if (!filter) return true;
return j.name.toLowerCase().includes(filter) || j.status.toLowerCase().includes(filter);
});

tbody.innerHTML = '';
filtered.forEach(j => {
const statusClass = j.status === 'success' ? '' : (j.status === 'fail' ? 'err' : (j.status === 'pending' ? 'warn' : 'paused'));
const badgeColor = statusClass === 'err' ? 'err' : (statusClass === 'warn' ? 'warn' : (statusClass === 'paused' ? 'paused' : ''));
const tr = document.createElement('tr');
tr.innerHTML = `<td>${j.name}</td>
<td><span class="status-indicator ${badgeColor}"></span>${j.status}</td>
<td>${j.last}</td>
<td><button class="action-btn" onclick="retryBuild('${j.id}')">rilancia</button></td>`;
tbody.appendChild(tr);
});

// aggiorna hero badge
const successCount = jobs.filter(j => j.status === 'success').length;
const failCount = jobs.filter(j => j.status === 'fail').length;
const otherCount = jobs.length - successCount - failCount;
let badgeText = `✅ ${successCount} ok`;
let badgeClass = '';
if (failCount > 0) { badgeText += ` · ${failCount} falliti`; badgeClass = 'err'; }
else if (otherCount > 0) { badgeText += ` · ${otherCount} in attesa`; badgeClass = 'warn'; }
const heroBadge = document.getElementById('heroBadge');
heroBadge.textContent = badgeText;
heroBadge.className = 'badge' + (badgeClass ? ' '+badgeClass : '');
}

// filtro
function applyFilter() { renderTable(); }

// rilancio singolo (simula cambiamento stato)
function retryBuild(jobId) {
const job = jobs.find(j => j.id === jobId);
if (!job) return;
// mostra popup conferma
showPopup(`rilanciare build "${job.name}"?`, () => {
job.status = 'pending';
job.last = new Date().toLocaleTimeString('it-IT', {hour:'2-digit',minute:'2-digit',second:'2-digit'});
renderTable();
closePopup();
});
}

// rilancio tutte
function retryAllBuilds() {
const failed = jobs.filter(j => j.status === 'fail' || j.status === 'paused');
if (failed.length === 0) { alert('nessuna build da rilanciare'); return; }
showPopup(`rilanciare ${failed.length} job in pausa/errore?`, () => {
failed.forEach(j => { j.status = 'pending'; j.last = new Date().toLocaleTimeString('it-IT', {hour:'2-digit',minute:'2-digit',second:'2-digit'}); });
renderTable();
closePopup();
});
}

// aggiungi job
function addJob() {
const input = document.getElementById('newJobName');
const name = input.value.trim();
if (!name) { alert('inserire un nome job'); return; }
// controlla duplicati (opzionale)
if (jobs.some(j => j.name === name)) { alert('job già esistente'); return; }
jobs.push({ id: 'job-'+Date.now(), name, status: 'pending', last: '--' });
input.value = '';
renderTable();
}

// popup system
function showPopup(msg, onConfirm) {
document.getElementById('popupMessage').textContent = msg;
document.getElementById('popupContainer').style.display = 'flex';
pendingPopupAction = onConfirm;
document.getElementById('popupConfirm').onclick = function() { if (pendingPopupAction) pendingPopupAction(); };
}

function closePopup() {
document.getElementById('popupContainer').style.display = 'none';
pendingPopupAction = null;
}

// init
renderTable();
</script>
