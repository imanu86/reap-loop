<script>
// ----- single-file vanilla JS -----
(function() {
  const tableBody = document.getElementById('tableBody');
  const filterInput = document.getElementById('filterInput');
  const heroStatus = document.getElementById('heroStatus');
  const relaunchAllBtn = document.getElementById('relaunchAllBtn');
  const addJobBtn = document.getElementById('addJobBtn');
  const newJobName = document.getElementById('newJobName');
  const popup = document.getElementById('confirmPopup');
  const popupMessage = document.getElementById('popupMessage');
  const popupCancel = document.getElementById('popupCancel');
  const popupConfirm = document.getElementById('popupConfirm');

  // pending action (relaunch single / relaunch all)
  let pendingAction = null;

  // ----- helper: update hero status -----
  function updateHeroStatus() {
    const rows = tableBody.querySelectorAll('tr');
    let ok = 0, fail = 0, pending = 0;
    rows.forEach(row => {
      const badge = row.querySelector('.status-badge');
      if (!badge) return;
      if (badge.classList.contains('ok')) ok++;
      else if (badge.classList.contains('fail')) fail++;
      else if (badge.classList.contains('pending')) pending++;
    });
    heroStatus.textContent = `● ${ok} OK · ${fail} FAIL · ${pending} PENDING`;
  }

  // ----- filter rows -----
  function applyFilter() {
    const q = filterInput.value.toLowerCase();
    const rows = tableBody.querySelectorAll('tr');
    rows.forEach(row => {
      const name = (row.dataset.name || '').toLowerCase();
      const idText = row.querySelector('.job-id')?.textContent || '';
      const match = name.includes(q) || idText.includes(q);
      row.style.display = match ? '' : 'none';
    });
  }

  // ----- relaunch single job (simulate) -----
  function relaunchJob(tr) {
    const badge = tr.querySelector('.status-badge');
    if (!badge) return;
    badge.className = 'status-badge pending';
    badge.textContent = 'PENDING';
    updateHeroStatus();
  }

  // ----- relaunch all visible -----
  function relaunchAllVisible() {
    const rows = tableBody.querySelectorAll('tr');
    rows.forEach(tr => {
      if (tr.style.display !== 'none') {
        const badge = tr.querySelector('.status-badge');
        if (badge) {
          badge.className = 'status-badge pending';
          badge.textContent = 'PENDING';
        }
      }
    });
    updateHeroStatus();
  }

  // ----- add new job row -----
  function addJob(name) {
    if (!name.trim()) return;
    const id = Date.now().toString(36).slice(-4);
    const nextId = tableBody.children.length + 1;
    const tr = document.createElement('tr');
    tr.dataset.id = id;
    tr.dataset.name = name.trim();
    tr.innerHTML = `<td>${name.trim()}</td><td><span class="job-id">#${String(nextId).padStart(3,'0')}</span></td><td><span class="status-badge pending">PENDING</span></td><td><button class="relaunchBtn">⟳ relaunch</button></td>`;
    tableBody.appendChild(tr);
    // attach relaunch event
    tr.querySelector('.relaunchBtn').addEventListener('click', function(e) {
      e.stopPropagation();
      const tr = this.closest('tr');
      if (tr) relaunchJob(tr);
    });
    updateHeroStatus();
    applyFilter(); // ensure filter hides if needed
  }

  // ----- popup handling -----
  function showPopup(message, action) {
    popupMessage.textContent = message;
    pendingAction = action;
    popup.style.display = 'flex';
  }

  popupCancel.addEventListener('click', function() {
    popup.style.display = 'none';
    pendingAction = null;
  });

  popupConfirm.addEventListener('click', function() {
    if (typeof pendingAction === 'function') pendingAction();
    popup.style.display = 'none';
    pendingAction = null;
  });

  // ----- event wiring -----
  // filter
  filterInput.addEventListener('input', applyFilter);

  // relaunch all button -> popup
  relaunchAllBtn.addEventListener('click', function() {
    showPopup('Relaunch all visible builds?', function() {
      relaunchAllVisible();
    });
  });

  // add job
  addJobBtn.addEventListener('click', function() {
    const name = newJobName.value;
    if (!name.trim()) return;
    addJob(name);
    newJobName.value = '';
  });

  // per-row relaunch buttons (initial + dynamic)
  document.addEventListener('click', function(e) {
    const btn = e.target.closest('.relaunchBtn');
    if (!btn) return;
    e.preventDefault();
    const tr =
