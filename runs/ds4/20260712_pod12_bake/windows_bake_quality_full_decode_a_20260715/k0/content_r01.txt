<script>
// ----- in-memory data -----
let jobs = [
  { id: 'j1', name: 'frontend', status: 'ok', branch: 'main' },
  { id: 'j2', name: 'api-gateway', status: 'fail', branch: 'develop' },
  { id: 'j3', name: 'worker', status: 'pending', branch: 'feat/cache' },
  { id: 'j4', name: 'auth-service', status: 'ok', branch: 'main' }
];
let nextId = 100;

// ----- DOM refs -----
const tbody = document.getElementById('tableBody');
const filterInput = document.getElementById('filterInput');
const resetBtn = document.getElementById('resetFilter');
const heroStatus = document.getElementById('heroStatus');
const jobCountSpan = document.getElementById('jobCount');
const newName = document.getElementById('newJobName');
const newBranch = document.getElementById('newJobBranch');
const addBtn = document.getElementById('addJobBtn');
const popupOverlay = document.getElementById('popupOverlay');
const popupMsg = document.getElementById('popupMessage');
const confirmBtn = document.getElementById('popupConfirm');
const cancelBtn = document.getElementById('popupCancel');

// ----- pending rebuild target (job id) -----
let pendingRebuildId = null;

// ----- helpers -----
function statusBadge(status) {
  const map = { ok: 'ok', fail: 'fail', pending: 'pending' };
  return `<span class="status-badge ${map[status] || 'pending'}">${status}</span>`;
}

function renderRows(filterText = '') {
  const f = filterText.toLowerCase().trim();
  const filtered = jobs.filter(j => {
    if (!f) return true;
    return j.name.toLowerCase().includes(f) || j.status.toLowerCase().includes(f);
  });
  if (filtered.length === 0) {
    tbody.innerHTML = `<tr><td colspan="4" style="text-align:center; padding:1.5rem; color:#64748b;">no matching jobs</td></tr>`;
  } else {
    tbody.innerHTML = filtered.map(j => `
      <tr>
        <td><strong>${j.name}</strong></td>
        <td>${statusBadge(j.status)}</td>
        <td>${j.branch}</td>
        <td><button class="rebuild-btn" data-id="${j.id}">🔄 relaunch</button></td>
      </tr>
    `).join('');
  }
  // attach rebuild events
  document.querySelectorAll('.rebuild-btn').forEach(btn => {
    btn.addEventListener('click', function(e) {
      const id = this.dataset.id;
      const job = jobs.find(j => j.id === id);
      if (!job) return;
      pendingRebuildId = id;
      popupMsg.textContent = `Rebuild "${job.name}" (${job.branch})?`;
      popupOverlay.style.display = 'flex';
    });
  });
  updateHeroAndCount();
}

function updateHeroAndCount() {
  const total = jobs.length;
  jobCountSpan.textContent = total;
  const okCount = jobs.filter(j => j.status === 'ok').length;
  const failCount = jobs.filter(j => j.status === 'fail').length;
  let text = '✅ all stable';
  if (failCount > 0) text = '⚠️ some failing';
  else if (okCount === 0 && total > 0) text = '⏳ pending';
  heroStatus.textContent = text;
}

// ----- add job -----
function addJob() {
  const name = newName.value.trim();
  const branch = newBranch.value.trim() || 'main';
  if (!name) return;
  const statuses = ['ok', 'fail', 'pending'];
  const randStatus = statuses[Math.floor(Math.random() * 3)];
  jobs.push({ id: 'j' + nextId++, name, status: randStatus, branch });
  renderRows(filterInput.value);
  newName.value = '';
  newBranch.value = 'main';
}

// ----- rebuild (relaunch) -----
function confirmRebuild() {
  if (!pendingRebuildId) return;
  const idx = jobs.findIndex(j => j.id === pendingRebuildId);
  if (idx === -1) return;
  // toggle status to pending (simulate relaunch)
  jobs[idx].status = 'pending';
  renderRows(filterInput.value);
  pendingRebuildId = null;
  popupOverlay.style.display = 'none';
}

// ----- cancel popup -----
function cancelPopup() {
  pendingRebuildId = null;
  popupOverlay.style.display = 'none';
}

// ----- filter & reset -----
filterInput.addEventListener('input', () => renderRows(filterInput.value));
resetBtn.addEventListener('click', () => {
  filterInput.value = '';
  renderRows('');
});

// ----- add event -----
addBtn.addEventListener('click', addJob);

// ----- popup events -----
confirmBtn.addEventListener('click', confirmRebuild);
cancelBtn.addEventListener('click', cancelPopup);
popupOverlay.addEventListener('click', function(e) {
  if (e.target === popupOverlay) cancelPopup();
});

// ----- initial render -----
renderRows('');
</script>
