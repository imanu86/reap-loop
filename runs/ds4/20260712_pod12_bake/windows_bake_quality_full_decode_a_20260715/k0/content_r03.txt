    <script>
        (function(){
            // stato interno
            let jobs = [
                { name: 'frontend-ci', status: 'success', branch: 'main', run: '2 min fa' },
                { name: 'api-tests', status: 'fail', branch: 'develop', run: '15 min fa' },
                { name: 'deploy-prod', status: 'success', branch: 'release', run: '1 ora fa' }
            ];
            let filterText = '';
            let pendingConfirmAction = null; // { type, index }

            // DOM refs
            const tbody = document.getElementById('tableBody');
            const filterInput = document.getElementById('filterInput');
            const filterBtn = document.getElementById('filterBtn');
            const newName = document.getElementById('newJobName');
            const newBranch = document.getElementById('newJobBranch');
            const addBtn = document.getElementById('addJobBtn');
            const popupOverlay = document.getElementById('popupOverlay');
            const popupMsg = document.getElementById('popupMsg');
            const popupConfirm = document.getElementById('popupConfirm');
            const popupCancel = document.getElementById('popupCancel');
            const navSpan = document.querySelector('nav span');
            const heroStatus = document.querySelector('.hero .status');

            // helper aggiornamento hero e nav
            function updateCounts() {
                const total = jobs.length;
                const success = jobs.filter(j => j.status === 'success').length;
                const fail = jobs.filter(j => j.status === 'fail').length;
                const pending = jobs.filter(j => j.status === 'pending').length;
                navSpan.textContent = total + ' job attivi';
                heroStatus.innerHTML = `
                    <div>✅ ${success} success</div>
                    <div>❌ ${fail} fail</div>
                    <div>⏳ ${pending} pending</div>
                `;
            }

            // render tabella con filtro
            function renderTable() {
                const filtered = jobs.filter(j => 
                    j.name.toLowerCase().includes(filterText.toLowerCase()) ||
                    j.branch.toLowerCase().includes(filterText.toLowerCase())
                );
                let html = '';
                filtered.forEach((job, idx) => {
                    const statusClass = job.status === 'success' ? 'success' : (job.status === 'fail' ? 'fail' : 'pending');
                    html += `<tr><td>${job.name}</td><td><span class="status-badge ${statusClass}">${job.status}</span></td><td>${job.branch}</td><td>${job.run}</td><td class="actions"><button class="rebuild" data-index="${idx}">🔄 rilancia</button><button class="remove" data-index="${idx}">✖ rimuovi</button></td></tr>`;
                });
                tbody.innerHTML = html;
                // attacca eventi ai bottoni (delega)
                tbody.querySelectorAll('.rebuild').forEach(btn => {
                    btn.addEventListener('click', function(e){
                        const idx = this.dataset.index;
                        const realIdx = jobs.findIndex(j => j.name === filtered[idx].name); // mapping corretto
                        if(realIdx === -1) return;
                        pendingConfirmAction = { type: 'rebuild', index: realIdx };
                        popupMsg.textContent = `Rilanciare la build "${jobs[realIdx].name}"?`;
                        popupOverlay.style.display = 'flex';
                    });
                });
                tbody.querySelectorAll('.remove').forEach(btn => {
                    btn.addEventListener('click', function(e){
                        const idx = this.dataset.index;
                        const realIdx = jobs.findIndex(j => j.name === filtered[idx].name);
                        if(realIdx === -1) return;
                        pendingConfirmAction = { type: 'remove', index: realIdx };
                        popupMsg.textContent = `Rimuovere definitivamente "${jobs[realIdx].name}"?`;
                        popupOverlay.style.display = 'flex';
                    });
                });
            }

            // applica filtro
            function applyFilter() {
                filterText = filterInput.value.trim();
                renderTable();
            }

            // aggiungi job
            function addJob() {
                const name = newName.value.trim();
                const branch = newBranch.value.trim();
                if(!name || !branch) { alert('Compila nome e branch'); return; }
                jobs.push({ name, status: 'pending', branch, run: 'ora' });
                newName.value = ''; newBranch.value = '';
                updateCounts();
                renderTable();
            }

            // gestione popup
            function confirmAction() {
                if(!pendingConfirmAction) return;
                const { type, index } = pendingConfirmAction;
                if(type === 'rebuild') {
                    jobs[index].status = 'pending';
                    jobs[index].run = 'adesso';
                } else if (type === 'remove') {
                    jobs.splice(index, 1);
                }
                pendingConfirmAction = null;
                updateCounts();
                renderTable();
                popupOverlay.style.display = 'none';
            }

            function cancelPopup() {
                pendingConfirmAction = null;
                popupOverlay.style.display = 'none';
            }

            // eventi
