<script>
// ---------- DATI INIZIALI ----------
let jobs = [
  { id: 'j1', name: 'frontend-ci', status: 'success', lastRun: '2 min fa' },
  { id: 'j2', name: 'api-test', status: 'fail', lastRun: '15 min fa' },
  { id: 'j3', name: 'deploy-prod', status: 'pending', lastRun: '1 h fa' },
  { id: 'j4', name: 'lint-check', status: 'success', lastRun: '5 min fa' }
];
let nextId = 5;

// riferimenti DOM
const tbody = document.getElementById('tableBody');
const heroStatus = document.getElementById('heroStatus');
const heroCount = document.getElementById('heroCount');
const heroLast = document.getElementById('heroLast');
const filterInput = document.getElementById('filterInput');
const filterBtn = document.getElementById('filterBtn');
const newJobName = document.getElementById('newJobName');
const addBtn = document.getElementById('addJobBtn');
const popupOverlay = document.getElementById('popupOverlay');
const popupMsg = document.getElementById('popupMsg');
const popupConfirm = document.getElementById('popupConfirm');
const popupCancel = document.getElementById('popupCancel');

// stato popup
let pendingRebuildId = null;

// ---------- RENDER ----------
function render(filterText = '') {
  const f = filterText.trim().toLowerCase();
  const filtered = f ? jobs.filter(j => j.name.toLowerCase().includes(f) || j.status.toLowerCase().includes(f)) : jobs;

  // righe tabella
  tbody.innerHTML = filtered.map(j => `
    <tr>
      <td><strong>${j.name}</strong></td>
      <td><span class="status-indicator ${j.status}"></span>${j.status}</td>
      <td>${j.lastRun}</td>
      <td><button class="action-btn" data-rebuild-id="${j.id}">🔄 rilancia</button></td>
    </tr>
  `).join('');

  // hero
  const successCount = jobs.filter(j => j.status === 'success').length;
  const failCount = jobs.filter(j => j.status === 'fail').length;
  const pendingCount = jobs.filter(j => j.status === 'pending').length;
  const total = jobs.length;
  heroCount.textContent = `${total} job (✅${successCount} ❌${failCount} ⏳${pendingCount})`;

  // ultimo job (più recente in base a lastRun, approssimato)
  const lastJob = jobs.reduce((prev, curr) => {
    // semplice euristica: se contiene 'min' più piccolo, etc. usiamo ordine array
    return prev ? prev : curr;
  }, null);
  heroLast.textContent = lastJob ? `${lastJob.name} — ${lastJob.lastRun}` : '—';

  // stato generale hero
  if (failCount > 0) {
    heroStatus.textContent = '❌ qualche fallimento';
    heroStatus.style.background = '#ef4444';
  } else if (pendingCount > 0) {
    heroStatus.textContent = '⏳ pending in corso';
    heroStatus.style.background = '#f59e0b';
  } else {
    heroStatus.textContent = '✅ tutto ok';
    heroStatus.style.background = '#22c55e';
  }

  // attach eventi ai bottoni rebuild (dopo render)
  document.querySelectorAll('[data-rebuild-id]').forEach(btn => {
    btn.addEventListener('click', function(e) {
      const id = this.dataset.rebuildId;
      const job = jobs.find(j => j.id === id);
      if (!job) return;
      pendingRebuildId = id;
      popupMsg.textContent = `Rilanciare build "${job.name}" ?`;
      popupOverlay.style.visibility = 'visible';
      popupOverlay.style.opacity = '1';
    });
  });
}

// ---------- POPUP ----------
popupCancel.addEventListener('click', () => {
  popupOverlay.style.visibility = 'hidden';
  popupOverlay.style.opacity = '0';
  pendingRebuildId = null;
});

popupConfirm.addEventListener('click', () => {
  if (pendingRebuildId) {
    const job = jobs.find(j => j.id === pendingRebuildId);
    if (job) {
      // simula rilancia: resetta stato a pending e aggiorna lastRun
      job.status = 'pending';
      job.lastRun = 'ora stessa';
      render(filterInput.value);
    }
    pendingRebuildId = null;
  }
  popupOverlay.style.visibility = 'hidden';
  popupOverlay.style.opacity = '0';
});

// ---------- FILTRO ----------
filterBtn.addEventListener('click', () => render(filterInput.value));
filterInput.addEventListener('keyup', (e) => { if (e.key === 'Enter') render(filterInput.value); });

// ---------- AGGIUNGI JOB ----------
addBtn.addEventListener('click', () => {
  const name = newJobName.value.trim();
  if (!name) return;
  jobs.push({
    id: 'j' + nextId++,
    name: name,
    status: 'pending',
    lastRun: 'appena aggiunto'
  });
  newJobName.value = '';
  render(filterInput.value);
});

// ---------- INIT ----------
render('');
</script>
