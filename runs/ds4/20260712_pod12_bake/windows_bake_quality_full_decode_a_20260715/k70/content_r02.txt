        (function() {

            let jobs = [

                { id: 'j3', name: 'deploy-backend', status: 'pending' },

            ];
            let selectedIds = new Set();

            // DOM refs
            const tbody = document.getElementById('tbody');
            const filterInput = document.getElementById('filterInput');
            const selectAllBox = document.getElementById('selectAll');
            const heroStatus = document.getElementById('heroStatus');
            const restartBtn = document.getElementById('restartSelected');
            const addJobBtn = document.getElementById('addJobBtn');
            const newJobName = document.getElementById('newJobName');
            const newJobStatus = document.getElementById('newJobStatus');
            const popupOverlay = document.getElementById('popupOverlay');
            const popupMessage = document.getElementById('popupMessage');
            const popupConfirm = document.getElementById('popupConfirm');
            const popupCancel = document.getElementById('popupCancel');

            // ---------- RENDER ----------
            function render() {
                const filter = filterInput.value.toLowerCase().trim();
                let filtered = jobs;
                if (filter) {
                    filtered = jobs.filter(j => 
                        j.name.toLowerCase().includes(filter) || 
                        j.status.toLowerCase().includes(filter)
                    );
                }

                // aggiorna stato hero
                const successCount = jobs.filter(j => j.status === 'success').length;
                const failCount = jobs.filter(j => j.status === 'fail').length;
                const pendingCount = jobs.filter(j => j.status === 'pending').length;
                heroStatus.textContent = `✅ ${successCount} success · ❌ ${failCount} fail · ⏳ ${pendingCount} pending`;

                // righe tabella
                let html = '';
                for (let job of filtered) {
                    const checked = selectedIds.has(job.id) ? 'checked' : '';
                    const badgeClass = job.status === 'success' ? 'badge success' : (job.status === 'fail' ? 'badge fail' : 'badge pending');
                    html += `<tr>
                        <td><input type="checkbox" class="row-check" data-id="${job.id}" ${checked}></td>
                        <td class="job-name">${job.name}</td>
                        <td><span class="${badgeClass}">${job.status}</span></td>
                        <td><button class="action-btn" data-action="restart" data-id="${job.id}">⟳ rilancia</button></td>
                    </tr>`;
                }
                tbody.innerHTML = html;

                // sincronizza selectAll
                const visibleRows = tbody.querySelectorAll('.row-check');
                const checkedRows = tbody.querySelectorAll('.row-check:checked');
                if (visibleRows.length === 0) {
                    selectAllBox.checked = false;
                } else {
                    selectAllBox.checked = (visibleRows.length === checkedRows.length);
                }

                // aggancia eventi riga (checkbox + pulsanti)
                tbody.querySelectorAll('.row-check').forEach(cb => {
                    cb.addEventListener('change', function(e) {
                        const id = this.dataset.id;
                        if (this.checked) selectedIds.add(id);
                        else selectedIds.delete(id);
                        // aggiorna selectAll
                        const all = tbody.querySelectorAll('.row-check');
                        const checkedAll = tbody.querySelectorAll('.row-check:checked');
                        selectAllBox.checked = (all.length === checkedAll.length);
                    });
                });

                tbody.querySelectorAll('[data-action="restart"]').forEach(btn => {
                    btn.addEventListener('click', function(e) {
                        const id = this.dataset.id;
                        // popup per singolo rilancio
                        const job = jobs.find(j => j.id === id);
                        if (!job) return;
                        popupMessage.textContent = `Rilanciare build "${job.name}" ?`;
                        popupOverlay.style.display = 'flex';
                        // salva riferimento
                        popupConfirm._pendingAction = () => {
                            // cambia stato a pending
                            job.status = 'pending';
                            render();
                            popupOverlay.style.display = 'none';
                        };
                        popupCancel.onclick = function() { popupOverlay.style.display = 'none'; };
                        popupConfirm.onclick = function() {
                            if (popupConfirm._pendingAction) popupConfirm._pendingAction();
                        };
                    });
                });
            }

            // ---------- EVENTI GLOBALI ----------
            // selectAll
            selectAllBox.addEventListener('change', function(e) {
                const checked = this.checked;
                const checkboxes = tbody.querySelectorAll('.row-check');
                checkboxes.forEach(cb => {
                    cb.checked = checked;
                    const id = cb.dataset.id;
                    if (checked) selectedIds.add(id);
                    else selectedIds.delete(id);
                });
            });

            // filtro
            filterInput.addEventListener('input', function() {
                render();
            });

            // pulsante rilancio selezionati (con popup)
            restartBtn.addEventListener('click', function() {
                if (selectedIds.size === 0) {
                    alert('Nessun job selezionato.');
                    return;
                }
                const names = jobs.filter(j => selectedIds.has(j.id)).map(j => j.name).join(', ');
                popupMessage.textContent = `Rilanciare ${selectedIds.size} build: ${names} ?`;
                popupOverlay.style.display = 'flex';
                popupConfirm._pendingAction = () => {
                    jobs.forEach(j => {
                        if (selectedIds.has(j.id)) j.status = 'pending';
                    });
                    selectedIds.clear();
                    render();
                    popupOverlay.style.display = 'none';
                };
                popupCancel.onclick = function() { popupOverlay.style.display = 'none'; };
                popupConfirm.onclick = function() {
                    if (popupConfirm._pendingAction) popupConfirm._pendingAction();
                };
            });
