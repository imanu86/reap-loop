        // ---------- DATI INIZIALI ----------
        const jobsData = [
            { name: 'frontend-ci', build: '#7821', status: 'success' },
            { name: 'api-tests', build: '#8912', status: 'fail' },
            { name: 'e2e-pipeline', build: '#9033', status: 'pending' },
            { name: 'lint-scan', build: '#452', status: 'success' },
            { name: 'deploy-staging', build: '#210', status: 'success' }
        ];

        let jobIdCounter = 100;
        let jobs = jobsData.map((j, idx) => ({ id: idx+1, name: j.name, build: j.build, status: j.status }));

        // ---------- RENDER ----------
        function renderTable() {
            const filterVal = (document.getElementById('filterInput').value || '').toLowerCase();
            const tbody = document.getElementById('jobTableBody');
            tbody.innerHTML = '';
            const filtered = jobs.filter(j => j.name.toLowerCase().includes(filterVal));
            if (filtered.length === 0) {
                tbody.innerHTML = `<tr><td colspan="5" style="text-align:center; padding:24px; color:#94a3b8;">nessun job corrisponde</td></tr>`;
                return;
            }
            filtered.forEach(j => {
                const badgeClass = j.status === 'success' ? 'badge ok' : (j.status === 'fail' ? 'badge fail' : 'badge pending');
                const badgeText = j.status === 'success' ? '✅ success' : (j.status === 'fail' ? '❌ fail' : '⏳ pending');
                const row = document.createElement('tr');
                row.dataset.jobId = j.id;
                row.innerHTML = `
                    <td><input type="checkbox" class="job-checkbox" data-id="${j.id}" onchange="updateSelectAll()"></td>
                    <td class="job-name">${j.name}</td>
                    <td>${j.build}</td>
                    <td><span class="${badgeClass}">${badgeText}</span></td>
                    <td><button class="action-btn" onclick="rebuildSingle(${j.id})">🚀 rilancia</button></td>
                `;
                tbody.appendChild(row);
            });
        }

        // ---------- FUNZIONI GLOBALI ----------
        window.applyFilter = renderTable;

        function updateSelectAll() {
            const checks = document.querySelectorAll('.job-checkbox');
            const allChecked = Array.from(checks).every(c => c.checked);
            document.getElementById('selectAll').checked = allChecked;
        }
        window.toggleAllCheckboxes = function() {
            const checked = document.getElementById('selectAll').checked;
            document.querySelectorAll('.job-checkbox').forEach(c => c.checked = checked);
        };

        // ---------- REBUILD (con popup) ----------
        let pendingRebuildIds = null;

        window.confirmRebuild = function() {
            const selected = Array.from(document.querySelectorAll('.job-checkbox:checked')).map(c => parseInt(c.dataset.id));
            if (selected.length === 0) {
                alert('Seleziona almeno un job.');
                return;
            }
            pendingRebuildIds = selected;
            const msg = document.getElementById('popupMsg');
            msg.textContent = `Rilanciare ${selected.length} build selezionate?`;
            document.getElementById('confirmPopup').style.display = 'flex';
        };

        window.closePopup = function() {
            document.getElementById('confirmPopup').style.display = 'none';
            pendingRebuildIds = null;
        };

        window.executeRebuild = function() {
            if (!pendingRebuildIds) return;
            pendingRebuildIds.forEach(id => {
                const job = jobs.find(j => j.id === id);
                if (job) job.status = 'pending';
            });
            pendingRebuildIds = null;
            closePopup();
            renderTable();
            updateHeroStatus();
        };

        // rebuild singolo (senza popup, ma cambia stato direttamente)
        window.rebuildSingle = function(id) {
            const job = jobs.find(j => j.id === id);
            if (job) job.status = 'pending';
            renderTable();
            updateHeroStatus();
        };

        // ---------- AGGIUNTA JOB ----------
        window.addJob = function() {
            const nameInput = document.getElementById('newJobName');
            const buildInput = document.getElementById('newBuildId');
            const name = nameInput.value.trim();
            const build = buildInput.value.trim();
            if (!name || !build) { alert('Compila nome e build ID'); return; }
            jobIdCounter++;
            jobs.push({ id: jobIdCounter, name: name, build: build, status: 'pending' });
