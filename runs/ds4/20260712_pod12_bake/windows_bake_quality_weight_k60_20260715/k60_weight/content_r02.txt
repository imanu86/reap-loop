    let jobs = [

    // --- render ---
    function renderTable() {
        const filterVal = document.getElementById('filterInput').value.toLowerCase();
        const tbody = document.getElementById('tableBody');
        let html = '';
        let successCount = 0, warningCount = 0, errorCount = 0;
        jobs.forEach(j => {
            if (filterVal && !j.name.toLowerCase().includes(filterVal) && !j.status.toLowerCase().includes(filterVal)) return;
            const statusClass = j.status === 'success' ? 'success' : j.status === 'fail' ? 'fail' : j.status === 'running' ? 'running' : 'pending';
            const badgeLabel = j.status === 'running' ? '⏳ running' : j.status === 'pending' ? 'pending' : j.status === 'success' ? '✔ success' : '✘ fail';
            if (j.status === 'success') successCount++;
            else if (j.status === 'running' || j.status === 'pending') warningCount++;
            else if (j.status === 'fail') errorCount++;
            html += `<tr>
                <td class="job-name">${j.name}</td>
                <td><span class="badge ${statusClass}">${badgeLabel}</span></td>
                <td>${j.build}</td>
                <td><button class="action-btn" data-id="${j.id}" onclick="confirmRelaunch('${j.id}')">▶️ rilancia</button></td>
            </tr>`;
        });
        tbody.innerHTML = html || `<tr><td colspan="4" style="text-align:center; padding:1.5rem; color:#888;">nessun job corrisponde</td></tr>`;

        // aggiorna hero status
        const hero = document.getElementById('heroStatus');
        let msg = `✅ ${successCount} OK`;
        if (warningCount > 0) msg += ` · ⚠️ ${warningCount} in corso`;
        if (errorCount > 0) msg += ` · ❌ ${errorCount} falliti`;
        hero.textContent = msg;
        hero.className = 'status';
        if (errorCount > 0) hero.classList.add('error');
        else if (warningCount > 0) hero.classList.add('warning');
    }

    function applyFilter() { renderTable(); }

    function refreshJobs() {
        // simulazione piccolo aggiornamento
        jobs.forEach(j => {
            if (j.status === 'running') j.status = 'success';
            else if (j.status === 'pending' && Math.random() > 0.6) j.status = 'running';
        });
        renderTable();
    }

    // --- popup ---
    let pendingJobId = null;
    function confirmRelaunch(jobId) {
        const job = jobs.find(j => j.id === jobId);
        if (!job) return;
        document.getElementById('popupJobName').textContent = `job: ${job.name}`;
        pendingJobId = jobId;
        document.getElementById('popupOverlay').style.display = 'flex';
        document.getElementById('confirmPopupBtn').onclick = function() {
            if (pendingJobId) {
                const j = jobs.find(j => j.id === pendingJobId);
                if (j) j.status = 'running';
                renderTable();
            }
            closePopup();
        };
    }

    function closePopup() {
        document.getElementById('popupOverlay').style.display = 'none';
        pendingJobId = null;
    }
