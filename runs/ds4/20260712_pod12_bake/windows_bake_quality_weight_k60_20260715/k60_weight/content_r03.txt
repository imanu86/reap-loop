<script>
(function(){
  const jobs = [
    { name:'Frontend', status:'✅ completato' },
    { name:'API-Gateway', status:'⚠️ in pausa' },
    { name:'Auth-Service', status:'❌ fallito' }
  ];
  let filterText = '';
  const tableBody = document.getElementById('tableBody');
  const filterInput = document.getElementById('filterInput');
  const resetBtn = document.getElementById('resetFilter');
  const heroStatus = document.getElementById('heroStatus');
  const heroTime = document.getElementById('heroTime');
  const newJobName = document.getElementById('newJobName');
  const addBtn = document.getElementById('addJobBtn');
  const popupRoot = document.getElementById('popupRoot');

  function setTime(){ heroTime.textContent = new Date().toLocaleTimeString(); }
  setTime();

  function updateHeroStatus(){
    const failed = jobs.some(j => j.status.includes('❌'));
    const pending = jobs.some(j => j.status.includes('⚠️'));
    if(failed) heroStatus.textContent = '❌ Problemi rilevati';
    else if(pending) heroStatus.textContent = '⚠️ Job in pausa';
    else heroStatus.textContent = '✅ Operativo';
  }

  function render(){
    const filtered = jobs.filter(j => {
      if(!filterText) return true;
      const q = filterText.toLowerCase();
      return j.name.toLowerCase().includes(q) || j.status.toLowerCase().includes(q);
    });
    if(filtered.length === 0){
      tableBody.innerHTML = '<tr><td colspan="3" style="text-align:center;padding:1rem;color:#6b7280">Nessun job corrisponde</td></tr>';
      return;
    }
    let html = '';
    filtered.forEach(j => {
      const btn = `<button class="action-btn" data-job="${j.name}">⟳ rilancia</button>`;
      html += `<tr><td>${j.name}</td><td>${j.status}</td><td>${btn}</td></tr>`;
    });
    tableBody.innerHTML = html;
    // attach listeners to all action buttons
    document.querySelectorAll('.action-btn').forEach(b => {
      b.addEventListener('click', function(e){
        const jobName = this.dataset.job;
        showConfirmPopup(jobName);
      });
    });
  }

  function showConfirmPopup(jobName){
    popupRoot.classList.remove('hidden');
    popupRoot.innerHTML = `
      <div class="popup-overlay">
        <div class="popup-box">
          <p>Rilanciare la build per <strong>${jobName}</strong>?</p>
          <button class="popup-confirm" id="confirmBtn">Sì, rilancia</button>
          <button class="popup-cancel" id="cancelBtn">Annulla</button>
        </div>
      </div>
    `;
    document.getElementById('confirmBtn').addEventListener('click', function(){
      const job = jobs.find(j => j.name === jobName);
      if(job){
        job.status = '⏳ in coda';
        updateHeroStatus();
        render();
      }
      popupRoot.classList.add('hidden');
    });
    document.getElementById('cancelBtn').addEventListener('click', function(){
      popupRoot.classList.add('hidden');
    });
    // click outside popup-box -> close
    document.querySelector('.popup-overlay').addEventListener('click', function(e){
      if(e.target === this) popupRoot.classList.add('hidden');
    });
  }

  filterInput.addEventListener('input', function(){
    filterText = this.value;
    render();
  });
  resetBtn.addEventListener('click', function(){
    filterInput.value = '';
    filterText = '';
    render();
  });

  addBtn.addEventListener('click', function(){
    const name = newJobName.value.trim();
    if(!name) return;
    if(jobs.some(j => j.name === name)){
      alert('Job già esistente');
      return;
    }
    jobs.push({ name, status:'✅ completato' });
    newJobName.value = '';
    updateHeroStatus();
    render();
  });

  // initial render
  render();
  updateHeroStatus();

  // attach to existing action buttons (after render)
  document.querySelectorAll('.action-btn').forEach(b => {
    b.addEventListener('click', function(){
      const jobName = this.dataset.job;
      showConfirmPopup(jobName);
    });
  });

  // extra: close popup via escape
  document.addEventListener('keydown', function(e){
    if(e.key === 'Escape') popupRoot.classList.add('hidden');
  });
})();
</script>
